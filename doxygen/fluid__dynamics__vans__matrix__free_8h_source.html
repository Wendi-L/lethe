13/**
14 * @brief A solver for the Volume-Averaged incompressible Navier-Stokes equations
15 * implemented in a matrix-free fashion.
16 *
17 * A matrix-free stabilized solver for the Volume-Averaged incompressible
18 * Navier-Stokes (VANS) equations implemented in a matrix-free fashion with a
19 * sum-factorization approach. It uses a continuous Galerkin discretization and
20 * solves the fully-coupled discretized problem in a monolithic way using
21 * multigrid preconditioners. This class greatly inherits from the matrix-free
22 * implementation of the Navier-Stokes equations.
23 *
24 * @tparam dim An integer that denotes the dimension of the space in which
25 * the flow is solved.
26 */
27template <int dim>
28class FluidDynamicsVANSMatrixFree : public FluidDynamicsMatrixFree<dim>
29{
30public:
31 /**
32 * @brief Constructor that sets the finite element degree and system operator
33 * according to simulation parameters. It initializes the CFD-DEM parameters,
34 * the particle mapping, the particle handler, and the void fraction manager.
35 * It also checks whether there are periodic boundaries.
36 *
37 * @param[in] param Relevant parameters for the solver.
38 */
39 FluidDynamicsVANSMatrixFree(CFDDEMSimulationParameters<dim> &param);
40
41 /**
42 * @brief Solve the problem defined by simulation parameters by iterating
43 * through time or through the mesh refinements.
44 */
45 virtual void
46 solve() override;
47
48protected:
49 /**
50 * @brief Setup the degree of freedom of the flow solver and the degrees of freedom
51 * and constraints for the void fraction.
52 */
53 virtual void
54 setup_dofs() override;
55
56 /**
57 * @brief Finish the time-step and manage the time-step end for the void
58 * fraction.
59 */
60 virtual void
61 finish_time_step_fd();
62
63 /**
64 * @brief Add data vectors to the data_out object for post_processing
65 * additional results. In this case, the void fraction field is added.
66 */
67 virtual void
68 output_field_hook(DataOut<dim> &data_out) override;
69
70 /// Simulation parameters for CFD-DEM simulations
71 CFDDEMSimulationParameters<dim> cfd_dem_simulation_parameters;
72
73 /// Mapping used for the particles
74 MappingQGeneric<dim> particle_mapping;
75
76 /// Particle-handler used to store particles in CFD-DEM simulations
77 Particles::ParticleHandler<dim, dim> particle_handler;
78
79 /// Object that manages the void fraction calculation from functions
80 /// or from parameters.
81 VoidFractionBase<dim> void_fraction_manager;
82
83
84 /// Member variables which are used to manage boundary conditions
85 bool has_periodic_boundaries;
86 Tensor<1, dim> periodic_offset;
87 unsigned int periodic_direction;
88};
