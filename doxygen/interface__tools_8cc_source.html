6template <int dim>
7double
8InterfaceTools::compute_cell_wise_volume(
9 FEPointEvaluation<1, dim> &fe_point_evaluation,
10 const typename DoFHandler<dim>::active_cell_iterator &cell,
11 Vector<double> cell_dof_level_set_values,
12 const double corr,
13 const unsigned int n_quad_points)
14{
15 const unsigned int n_dofs = cell_dof_level_set_values.size();
16
17 // Initialize required variables to compute local volume
18 const BoundingBox<dim> unit_box = create_unit_bounding_box<dim>();
19 CellWiseFunction<dim> signed_distance_function =
20 CellWiseFunction<dim>(cell->get_fe().degree);
21
22 hp::QCollection<1> q_collection;
23 q_collection.push_back(QGauss<1>(n_quad_points));
24
25 NonMatching::QuadratureGenerator<dim> quadrature_generator =
26 NonMatching::QuadratureGenerator<dim>(q_collection);
27
28 for (unsigned int j = 0; j < n_dofs; j++)
29 {
30 cell_dof_level_set_values[j] += corr;
31 }
32
33 signed_distance_function.set_active_cell(cell_dof_level_set_values);
34 quadrature_generator.generate(signed_distance_function, unit_box);
35
36 const Quadrature<dim> inside_quadrature =
37 quadrature_generator.get_inside_quadrature();
38
39 if (inside_quadrature.size() == 0)
40 return 0.0;
41
42 fe_point_evaluation.reinit(cell, inside_quadrature.get_points());
43
44 double inside_cell_volume = 0.0;
45 for (unsigned int q = 0; q < inside_quadrature.size(); q++)
46 {
47 /* Compute the volume int 1*JxW*dOmega. FEPointEvaluation.JxW() does not
48 return the right thing.*/
49 inside_cell_volume += fe_point_evaluation.jacobian(q).determinant() *
50 inside_quadrature.weight(q);
51 }
52
53 return inside_cell_volume;
54}

56template <int dim, typename VectorType>
57double
58InterfaceTools::compute_volume(const Mapping<dim> &mapping,
59 const DoFHandler<dim> &dof_handler,
60 const FiniteElement<dim> &fe,
61 const VectorType &level_set_vector,
62 const double iso_level,
63 const MPI_Comm &mpi_communicator)
64{
65 FEPointEvaluation<1, dim> fe_point_evaluation(
66 mapping, fe, update_jacobians | update_JxW_values);
67
68 double volume = 0.0;
69 for (const auto &cell : dof_handler.active_cell_iterators())
70 {
71 if (cell->is_locally_owned())
72 {
73 const unsigned int n_dofs_per_cell = cell->get_fe().n_dofs_per_cell();
74 Vector<double> cell_dof_level_set_values(n_dofs_per_cell);
75
76 cell->get_dof_values(level_set_vector,
77 cell_dof_level_set_values.begin(),
78 cell_dof_level_set_values.end());
79
80 const double level_set_correction = -iso_level;
81 volume += compute_cell_wise_volume(fe_point_evaluation,
82 cell,
83 cell_dof_level_set_values,
84 level_set_correction,
85 cell->get_fe().degree + 1);
86 }
87 }
88 volume = Utilities::MPI::sum(volume, mpi_communicator);
89
90 return volume;
91}

108template <int dim, typename VectorType>
109void
110InterfaceTools::reconstruct_interface(
111 const Mapping<dim> &mapping,
112 const DoFHandler<dim> &dof_handler,
113 const FiniteElement<dim> &fe,
114 const VectorType &level_set_vector,
115 const double iso_level,
116 std::map<types::global_cell_index, std::vector<Point<dim>>>
117 &interface_reconstruction_vertices,
118 std::map<types::global_cell_index, std::vector<CellData<dim - 1>>>
119 &interface_reconstruction_cells,
120 std::set<types::global_dof_index> &intersected_dofs)
121{
122 GridTools::MarchingCubeAlgorithm<dim, VectorType> marching_cube(mapping,
123 fe,
124 1,
125 1e-10);
126 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
127
128 for (const auto &cell : dof_handler.active_cell_iterators())
129 {
130 if (cell->is_locally_owned())
131 {
132 std::vector<Point<dim>> surface_vertices;
133 std::vector<CellData<dim - 1>> surface_cells;
134
135 marching_cube.process_cell(
136 cell, level_set_vector, iso_level, surface_vertices, surface_cells);
137
138 // If the cell is intersected, reconstruct the interface in it
139 if (surface_vertices.size() != 0)
140 {
141 const unsigned int cell_index = cell->global_active_cell_index();
142
143 // Store the interface reconstruction vertices and cells
144 interface_reconstruction_vertices[cell_index] = surface_vertices;
145 interface_reconstruction_cells[cell_index] = surface_cells;
146
147 // Store the dofs of the intersected volume cell
148 std::vector<types::global_dof_index> dof_indices(dofs_per_cell);
149 cell->get_dof_indices(dof_indices);
150
151 for (unsigned int i = 0; i < dofs_per_cell; ++i)
152 {
153 intersected_dofs.insert(dof_indices[i]);
154 }
155 }
156 }
157 }
158}

187template <int dim, typename VectorType>
188void
189InterfaceTools::SignedDistanceSolver<dim, VectorType>::setup_dofs()
190{
191 const MPI_Comm mpi_communicator = dof_handler.get_communicator();
192
193 dof_handler.distribute_dofs(*this->fe);
194
195 locally_owned_dofs = dof_handler.locally_owned_dofs();
196 locally_relevant_dofs = DoFTools::extract_locally_relevant_dofs(dof_handler);
197 locally_active_dofs = DoFTools::extract_locally_active_dofs(dof_handler);
198
199 level_set.reinit(locally_owned_dofs, locally_relevant_dofs, mpi_communicator);
200
201 signed_distance.reinit(locally_owned_dofs,
202 locally_active_dofs,
203 mpi_communicator);
204 signed_distance_with_ghost.reinit(locally_owned_dofs,
205 locally_active_dofs,
206 mpi_communicator);
207
208 distance.reinit(locally_owned_dofs, locally_active_dofs, mpi_communicator);
209 distance_with_ghost.reinit(locally_owned_dofs,
210 locally_active_dofs,
211 mpi_communicator);
212
213 volume_correction.reinit(locally_owned_dofs,
214 locally_active_dofs,
215 mpi_communicator);
216
217 constraints.clear();
218 constraints.reinit(locally_owned_dofs, locally_relevant_dofs);
219 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
220 constraints.close();
221}

250template <int dim, typename VectorType>
251void
252InterfaceTools::SignedDistanceSolver<dim, VectorType>::initialize_distance()
253{
254 /* Initialization of the active dofs to the max distance we want to
255 redistanciate. It requires a loop on the active dofs to initialize also
256 the distance value of the ghost dofs. Otherwise, they are set to 0.0 by
257 default. */
258 for (auto p : this->locally_active_dofs)
259 {
260 distance(p) = max_distance;
261 distance_with_ghost(p) = max_distance;
262
263 const double level_set_value = level_set(p);
264 signed_distance(p) = max_distance * sgn(level_set_value);
265 signed_distance_with_ghost(p) = max_distance * sgn(level_set_value);
266 }
267}

269template <int dim, typename VectorType>
270VectorType &
271InterfaceTools::SignedDistanceSolver<dim, VectorType>::get_signed_distance()
272{
273 const MPI_Comm mpi_communicator = dof_handler.get_communicator();
274
275 VectorType tmp_local_level_set(this->locally_owned_dofs, mpi_communicator);
276
277 // Loop on the DoFs to be compatible with the difference in vector type
278 // between level_set and signed_distance
279 for (auto p : this->locally_owned_dofs)
280 {
281 tmp_local_level_set(p) = signed_distance(p);
282 }
283
284 level_set = tmp_local_level_set;
285
286 return level_set;
287}
