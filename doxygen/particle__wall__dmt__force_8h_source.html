100  /**
101  * @brief Carries out calculation of the rolling resistance torque using the
102  * constant model
103  *
104  * @param particle_properties Particle properties
105  * @param effective_rolling_friction_coefficient Effective rolling friction
106  * coefficient
107  * @param normal_force_norm Normal force norm
108  * @return rolling resistance torque
109  */
110  inline Tensor<1, 3>
111  constant_resistance(const ArrayView<const double> &particle_properties,
112  const double effective_rolling_friction_coefficient,
113  const double normal_force_norm,
114  const Tensor<1, 3> & /*normal_contact_vector*/)
115  {
116  // Getting the angular velocity of particle in the vector format
117  Tensor<1, 3> angular_velocity;
118  for (int d = 0; d < 3; ++d)
119  {
120  angular_velocity[d] =
121  particle_properties[DEM::PropertiesIndex::omega_x + d];
122  }
123 
124  // Calculation of particle-wall angular velocity (norm of the
125  // particle angular velocity)
126  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
127 
128  double omega_value = angular_velocity.norm();
129  if (omega_value != 0)
130  {
131  particle_wall_angular_velocity = angular_velocity / omega_value;
132  }
133 
134  // Calculation of rolling resistance torque
135  Tensor<1, 3> rolling_resistance_torque =
136  -effective_rolling_friction_coefficient *
137  (particle_properties[DEM::PropertiesIndex::dp] * 0.5) *
138  normal_force_norm * particle_wall_angular_velocity;
139 
140  return rolling_resistance_torque;
141  }

143  /**
144  * @brief Carries out calculation of the rolling resistance torque using the
145  * viscous model
146  *
147  * @param particle_properties Particle properties
148  * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
149  * @param normal_force_norm Normal force norm
150  * @return rolling resistance torque
151  */
152  inline Tensor<1, 3>
153  viscous_resistance(const ArrayView<const double> &particle_properties,
154  const double effective_rolling_friction_coefficient,
155  const double normal_force_norm,
156  const Tensor<1, 3> &normal_contact_vector)
157  {
158  // Getting the angular velocity of particle in the vector format
159  Tensor<1, 3> angular_velocity;
160  for (int d = 0; d < 3; ++d)
161  {
162  angular_velocity[d] =
163  particle_properties[DEM::PropertiesIndex::omega_x + d];
164  }
165 
166  // Calculation of particle-wall angular velocity (norm of the
167  // particle angular velocity)
168  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
169 
170  double omega_value = angular_velocity.norm();
171  if (omega_value != 0)
172  {
173  particle_wall_angular_velocity = angular_velocity / omega_value;
174  }
175 
176  Tensor<1, 3> v_omega =
177  cross_product_3d(angular_velocity,
178  particle_properties[DEM::PropertiesIndex::dp] * 0.5 *
179  normal_contact_vector);
180 
181  // Calculation of rolling resistance torque
182  Tensor<1, 3> rolling_resistance_torque =
183  -effective_rolling_friction_coefficient *
184  particle_properties[DEM::PropertiesIndex::dp] * 0.5 * normal_force_norm *
185  v_omega.norm() * particle_wall_angular_velocity;
186 
187  return rolling_resistance_torque;
188  }

190  /**
191  * @brief Initialize every particle-wall properties required
192  * for the force calculation. This function is required for the DMT force
193  * model, since the dmt_cut_off_distance attribute needs the be set to const.
194  * To do so, the effective surface energy and of the effective hamaker contant
195  * need to be initialized before hand. For code clarity, we also initialized
196  * every other parameter, otherwise the initialization would take place at two
197  * different places in the code.
198  *
199  * @param[in] dem_parameters DEM parameters declared in the .prm file
200  */
201  void
202  initialize_particle_wall_properties(
203  const DEMSolverParameters<dim> &dem_parameters)
204  {
205  const double wall_youngs_modulus =
206  dem_parameters.lagrangian_physical_properties.youngs_modulus_wall;
207  const double wall_poisson_ratio =
208  dem_parameters.lagrangian_physical_properties.poisson_ratio_wall;
209  const double wall_restitution_coefficient =
210  dem_parameters.lagrangian_physical_properties
211  .restitution_coefficient_wall;
212  const double wall_friction_coefficient =
213  dem_parameters.lagrangian_physical_properties.friction_coefficient_wall;
214  const double wall_rolling_friction_coefficient =
215  dem_parameters.lagrangian_physical_properties.rolling_friction_wall;
216  const double wall_surface_energy =
217  dem_parameters.lagrangian_physical_properties.surface_energy_wall;
218  const double wall_hamaker_constant =
219  dem_parameters.lagrangian_physical_properties.hamaker_constant_wall;
220  for (unsigned int i = 0;
221  i < dem_parameters.lagrangian_physical_properties.particle_type_number;
222  ++i)
223  {
224  const double particle_youngs_modulus =
225  dem_parameters.lagrangian_physical_properties.youngs_modulus_particle
226  .at(i);
227  const double particle_poisson_ratio =
228  dem_parameters.lagrangian_physical_properties.poisson_ratio_particle
229  .at(i);
230  const double particle_restitution_coefficient =
231  dem_parameters.lagrangian_physical_properties
232  .restitution_coefficient_particle.at(i);
233  const double particle_friction_coefficient =
234  dem_parameters.lagrangian_physical_properties
235  .friction_coefficient_particle.at(i);
236  const double particle_rolling_friction_coefficient =
237  dem_parameters.lagrangian_physical_properties
238  .rolling_friction_coefficient_particle.at(i);
239  const double particle_surface_energy =
240  dem_parameters.lagrangian_physical_properties.surface_energy_particle
241  .at(i);
242  const double particle_hamaker_constant =
243  dem_parameters.lagrangian_physical_properties
244  .hamaker_constant_particle.at(i);
245 
246  this->effective_youngs_modulus[i] =
247  (particle_youngs_modulus * wall_youngs_modulus) /
248  (wall_youngs_modulus *
249  (1 - particle_poisson_ratio * particle_poisson_ratio) +
250  particle_youngs_modulus *
251  (1 - wall_poisson_ratio * wall_poisson_ratio) +
252  DBL_MIN);
253 
254  this->effective_shear_modulus[i] =
255  (particle_youngs_modulus * wall_youngs_modulus) /
256  ((2 * wall_youngs_modulus * (2 - particle_poisson_ratio) *
257  (1 + particle_poisson_ratio)) +
258  (2 * particle_youngs_modulus * (2 - wall_poisson_ratio) *
259  (1 + wall_poisson_ratio)) +
260  DBL_MIN);
261 
262  this->effective_coefficient_of_restitution[i] =
263  2 * particle_restitution_coefficient * wall_restitution_coefficient /
264  (particle_restitution_coefficient + wall_restitution_coefficient +
265  DBL_MIN);
266 
267  const double log_coeff_restitution =
268  log(this->effective_coefficient_of_restitution[i]);
269  this->model_parameter_beta[i] =
270  log_coeff_restitution /
271  sqrt((log_coeff_restitution * log_coeff_restitution) + 9.8696);
272 
273  this->effective_coefficient_of_friction[i] =
274  2 * particle_friction_coefficient * wall_friction_coefficient /
275  (particle_friction_coefficient + wall_friction_coefficient + DBL_MIN);
276 
277  this->effective_coefficient_of_rolling_friction[i] =
278  2 * particle_rolling_friction_coefficient *
279  wall_rolling_friction_coefficient /
280  (particle_rolling_friction_coefficient +
281  wall_rolling_friction_coefficient + DBL_MIN);
282 
283  this->effective_surface_energy[i] =
284  particle_surface_energy + wall_surface_energy -
285  std::pow(std::sqrt(particle_surface_energy) -
286  std::sqrt(wall_surface_energy),
287  2);
288  this->effective_hamaker_constant[i] =
289  0.5 * (particle_hamaker_constant + wall_hamaker_constant);
290  }
291  }

294  /**
295  * @brief This function uses the effective surface energy and effective hamaker
296  * constant to compute the cut off distance for the force calculation.
297  *
298  * @param[in] dem_parameters DEM parameters declared in the .prm file
299  *
300  * @return Cut off distance for the force calculation.
301  */
302  double
303  set_dmt_cut_off_distance()
304  {
305  // We are searching for the large delta_o, thus the smallest effective
306  // surface energy and largest Hamaker constant.
307  double max_effective_hamaker_constant =
308  *(std::max_element(this->effective_hamaker_constant.begin(),
309  this->effective_hamaker_constant.end()));
310  double min_effective_surface_energy =
311  *(std::min_element(this->effective_surface_energy.begin(),
312  this->effective_surface_energy.end()));
313 
314  const double delta_0 =
315  -std::sqrt(max_effective_hamaker_constant /
316  (12. * M_PI * min_effective_surface_energy));
317 
318  return delta_0 / std::sqrt(dmt_cut_off_threshold);
319  };
320
