27 /**
28  * Base class for the particles force chains contact force models.
29  * This class does not implement any of the models, but ensures that
30  * an interface without template specialization is available. All of the

32  * ParticlesForceChains class which is templated by the contact model
33  * type.
34  */

148 private:
149  /**
150  * @brief Execute the contact calculation step for the particle-particle
151  * contact only for local-local and local-ghost contacts with no periodicity.
152  * This is a simplified version of the contact calculation of the
153  * particle-particle contact forces class, without the other contact types and
154  * the update of the particles forces, torques and tangential overlap.
155  *
156  * @param[in] adjacent_particles_list Container of the adjacent particles of a
157  * particles.
158  */
159  inline void
160  execute_contact_calculation(
161  typename DEM::dem_data_structures<dim>::particle_contact_info
162  &adjacent_particles_list)
163  {
164  // No contact calculation if no adjacent particles
165  if (adjacent_particles_list.empty())
166  return;
167 
168  const double force_calculation_threshold_distance =
169  this->get_force_calculation_threshold_distance();
170 
171  // Define local variables which will be used within the contact calculation
172  Tensor<1, 3> normal_unit_vector;
173  Tensor<1, 3> normal_force;
174  Tensor<1, 3> tangential_force;
175  Tensor<1, 3> particle_one_tangential_torque;
176  Tensor<1, 3> particle_two_tangential_torque;
177  Tensor<1, 3> rolling_resistance_torque;
178  double normal_relative_velocity_value;
179  Tensor<1, 3> tangential_relative_velocity;
180 
181  // Gather information about particle 1 and set it up.
182  auto first_contact_info = adjacent_particles_list.begin();
183  auto particle_one = first_contact_info->second.particle_one;
184  auto particle_one_properties = particle_one->get_properties();
185 
186  // Fix particle one location for 2d and 3d
187  Point<3> particle_one_location = this->get_location(particle_one);
188 
189  for (auto &&contact_info :
190  adjacent_particles_list | boost::adaptors::map_values)
191  {
192  // Getting information (location and properties) of particle 2 in
193  // contact with particle 1
194  auto particle_two = contact_info.particle_two;
195  auto particle_two_properties = particle_two->get_properties();
196 
197  // Get particle 2 location in dimension independent way
198  Point<3> particle_two_location = this->get_location(particle_two);
199 
200  // Calculation of normal overlap
201  double normal_overlap =
202  0.5 * (particle_one_properties[PropertiesIndex::dp] +
203  particle_two_properties[PropertiesIndex::dp]) -
204  particle_one_location.distance(particle_two_location);
205 
206  if (normal_overlap > force_calculation_threshold_distance)
207  {
208  update_contact_information(tangential_relative_velocity,
209  normal_relative_velocity_value,
210  normal_unit_vector,
211  particle_one_properties,
212  particle_two_properties,
213  particle_one_location,
214  particle_two_location);
215 
216  this->calculate_contact(contact_info,
217  tangential_relative_velocity,
218  normal_relative_velocity_value,
219  normal_unit_vector,
220  normal_overlap,
221  particle_one_properties,
222  particle_two_properties,
223  normal_force,
224  tangential_force,
225  particle_one_tangential_torque,
226  particle_two_tangential_torque,
227  rolling_resistance_torque);
228  }
229 
230  vertices.push_back(particle_one_location);
231  vertices.push_back(particle_two_location);
232  force_normal.push_back(sqrt(normal_force.norm()));
233  }
234  }

236  /**
237  * @brief Update the contact pair information for all contact force
238  * calculations.
239  *
240  * @param[out] tangential_relative_velocity Tangential relative velocity.
241  * @param[out] normal_relative_velocity_value Normal relative velocity.
242  * @param[out] normal_unit_vector Normal vector of the contact.
243  * @param[in] particle_one_properties Properties of particle one in contact.
244  * @param[in] particle_two_properties Properties of particle two in contact.
245  * @param[in] particle_one_location Location of particle one in contact.
246  * @param[in] particle_two_location Location of particle two in contact.
247  */
248  inline void
249  update_contact_information(
250  Tensor<1, 3> &tangential_relative_velocity,
251  double &normal_relative_velocity_value,
252  Tensor<1, 3> &normal_unit_vector,
253  const ArrayView<const double> &particle_one_properties,
254  const ArrayView<const double> &particle_two_properties,
255  const Point<3> &particle_one_location,
256  const Point<3> &particle_two_location)
257  {
258  // Calculation of the contact vector from particle one to particle two
259  Tensor<1, 3> contact_vector = particle_two_location - particle_one_location;
260 
261  // Calculation of the normal unit contact vector
262  normal_unit_vector = contact_vector / contact_vector.norm();
263 
264  // Defining velocities and angular velocities of particles one and
265  // two as vectors
266  Tensor<1, 3> particle_one_omega, particle_two_omega;
267 
268  // Defining relative contact velocity
269  Tensor<1, 3> contact_relative_velocity;
270 
271  // Assigning velocities and angular velocities of particles
272  contact_relative_velocity[0] =
273  particle_one_properties[PropertiesIndex::v_x] -
274  particle_two_properties[PropertiesIndex::v_x];
275  contact_relative_velocity[1] =
276  particle_one_properties[PropertiesIndex::v_y] -
277  particle_two_properties[PropertiesIndex::v_y];
278  contact_relative_velocity[2] =
279  particle_one_properties[PropertiesIndex::v_z] -
280  particle_two_properties[PropertiesIndex::v_z];
281 
282  particle_one_omega[0] = particle_one_properties[PropertiesIndex::omega_x];
283  particle_one_omega[1] = particle_one_properties[PropertiesIndex::omega_y];
284  particle_one_omega[2] = particle_one_properties[PropertiesIndex::omega_z];
285 
286  particle_two_omega[0] = particle_two_properties[PropertiesIndex::omega_x];
287  particle_two_omega[1] = particle_two_properties[PropertiesIndex::omega_y];
288  particle_two_omega[2] = particle_two_properties[PropertiesIndex::omega_z];
289 
290  // Calculation of contact relative velocity
291  // v_ij = (v_i - v_j) + (R_i*omega_i + R_j*omega_j) × n_ij
292  contact_relative_velocity += (cross_product_3d(
293  0.5 * (particle_one_properties[PropertiesIndex::dp] * particle_one_omega +
294  particle_two_properties[PropertiesIndex::dp] * particle_two_omega),
295  normal_unit_vector));
296 
297  // Calculation of normal relative velocity. Note that in the
298  // following line the product acts as inner product since both
299  // sides are vectors, while in the second line the product is
300  // scalar and vector product
301  normal_relative_velocity_value =
302  contact_relative_velocity * normal_unit_vector;
303 
304  // Calculation of tangential relative velocity
305  // v_rt = v_ij - (v_ij⋅n_ij)*n_ij
306  tangential_relative_velocity =
307  contact_relative_velocity -
308  (normal_relative_velocity_value * normal_unit_vector);
309  }
310
