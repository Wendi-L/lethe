9 // This function finds all the particle-point contact candidates
10 template <int dim>
11 void
12 find_particle_point_contact_pairs(
13  const Particles::ParticleHandler<dim> &particle_handler,
14  const std::unordered_map<std::string, cell_point_info<dim>>
15  &boundary_cells_with_points,
16  typename DEM::dem_data_structures<dim>::particle_point_candidates
17  &particle_point_contact_candidates)
18 {
19  // Clear the candidate map
20  particle_point_contact_candidates.clear();
21 
22  // Defining and resetting a local particle-point candidate counter. This is
23  // used as a key to the output map
24  int contact_candidate_counter = 0;
25 
26  // Iterating over the boundary_cells_with_points which is the output of
27  // the find_boundary_cells_information class.
28  // This vector contains all the required information of the boundary
29  // cells with points. In this loop we find the particles located in each of
30  // these boundary cells with points
31  for (auto map_iterator = boundary_cells_with_points.begin();
32  map_iterator != boundary_cells_with_points.end();
33  ++map_iterator)
34  {
35  const cell_point_info<dim> &cells_with_boundary_points_information =
36  map_iterator->second;
37 
38  // Finding particles located in the corresponding cell
39  typename Particles::ParticleHandler<dim>::particle_iterator_range
40  particles_in_cell = particle_handler.particles_in_cell(
41  cells_with_boundary_points_information.cell);
42 
43  for (typename Particles::ParticleHandler<dim>::particle_iterator_range::
44  iterator particles_in_cell_iterator = particles_in_cell.begin();
45  particles_in_cell_iterator != particles_in_cell.end();
46  ++particles_in_cell_iterator)
47  {
48  // Making the pair and adding it to the
49  // particle_point_contact_candidates map. This map is the output of
50  // this function
51  particle_point_contact_candidates.emplace(
52  contact_candidate_counter,
53  particle_point_contact_info<dim>{
54  particles_in_cell_iterator,
55  cells_with_boundary_points_information.point});
56  ++contact_candidate_counter;
57  }
58  }
59 }
