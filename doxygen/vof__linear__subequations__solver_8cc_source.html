10 template <int dim, typename ScratchDataType>
11 void
12 VOFLinearSubequationsSolver<dim, ScratchDataType>::setup_dofs()
13 {
14  // Get MPI communicator
15  auto mpi_communicator = this->triangulation->get_communicator();
16 
17  // Distribute and renumber DoFs
18  this->dof_handler.distribute_dofs(*this->fe);
19  DoFRenumbering::Cuthill_McKee(this->dof_handler);
20 
21  // Get locally owned and relevant DoFs
22  this->locally_owned_dofs = this->dof_handler.locally_owned_dofs();
23  this->locally_relevant_dofs =
24  DoFTools::extract_locally_relevant_dofs(this->dof_handler);
25 
26  // Constraints
27  this->constraints.clear();
28  this->constraints.reinit(this->locally_relevant_dofs);
29  DoFTools::make_hanging_node_constraints(this->dof_handler, this->constraints);
30  this->constraints.close();
31 
32  // Sparsity pattern
33  DynamicSparsityPattern dsp(this->locally_relevant_dofs);
34  DoFTools::make_sparsity_pattern(this->dof_handler,
35  dsp,
36  this->constraints,
37  false);
38  SparsityTools::distribute_sparsity_pattern(dsp,
39  this->locally_owned_dofs,
40  mpi_communicator,
41  this->locally_relevant_dofs);
42 
43  // Reinitialize system matrix and right-hand side (rhs)
44  this->system_matrix.reinit(this->locally_owned_dofs,
45  this->locally_owned_dofs,
46  dsp,
47  mpi_communicator);
48  this->system_rhs.reinit(this->locally_owned_dofs, mpi_communicator);
49 
50  // Reinitialize solution vectors
51  this->present_solution.reinit(this->locally_owned_dofs,
52  this->locally_relevant_dofs,
53  mpi_communicator);
54  this->evaluation_point = this->present_solution;
55 
56 
57  if (this->subequation_verbosity != Parameters::Verbosity::quiet)
58  {
59  std::string subequation_string =
60  this->subequations->get_subequation_string(this->subequation_id);
61  this->pcout << " Number of " << subequation_string
62  << " degrees of freedom: " << this->dof_handler.n_dofs()
63  << std::endl;
64  }
65 
66  // Provide DoFHandler and solutions to the subequations interface
67  this->subequations->set_dof_handler(this->subequation_id, &this->dof_handler);
68  this->subequations->set_solution(this->subequation_id,
69  &this->present_solution);
70 }

166 template <int dim, typename ScratchDataType>
167 void
168 VOFLinearSubequationsSolver<dim, ScratchDataType>::
169  solve_linear_system_and_update_solution(const bool &is_post_mesh_adaptation)
170 {
171  auto mpi_communicator = this->triangulation->get_communicator();
172 
173  const AffineConstraints<double> &constraints_used = this->constraints;
174 
175  const bool verbose(
176  this->subequation_verbosity != Parameters::Verbosity::quiet &&
177  this->linear_solver_verbosity != Parameters::Verbosity::quiet &&
178  !is_post_mesh_adaptation);
179 
180  if (verbose)
181  {
182  std::string subequation_string =
183  this->subequations->get_subequation_string(this->subequation_id);
184 
185  this->pcout << " -Solving " << subequation_string << ":" << std::endl;
186  }
187 
188  // Set tolerance
189  const double linear_solver_tolerance =
190  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
191  .minimum_residual;
192 
193  // Solution vector
194  GlobalVectorType completely_distributed_solution(this->locally_owned_dofs,
195  mpi_communicator);
196  completely_distributed_solution = this->present_solution;
197 
198  if (verbose)
199  {
200  this->pcout << " -Tolerance of iterative solver is: "
201  << linear_solver_tolerance << std::endl;
202  }
203 
204  // ILU preconditioner
205  const double ilu_fill =
206  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
207  .ilu_precond_fill;
208  const double ilu_atol =
209  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
210  .ilu_precond_atol;
211  const double ilu_rtol =
212  this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
213  .ilu_precond_rtol;
214  TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
215  ilu_fill, ilu_atol, ilu_rtol, 0);
216 
217  this->ilu_preconditioner =
218  std::make_shared<TrilinosWrappers::PreconditionILU>();
219 
220  this->ilu_preconditioner->initialize(this->system_matrix,
221  preconditionerOptions);
222 
223  // CG solver
224  SolverControl solver_control(
225  this->simulation_parameters.linear_solver.at(PhysicsID::VOF).max_iterations,
226  linear_solver_tolerance,
227  true,
228  true);
229 
230  TrilinosWrappers::SolverCG solver(solver_control);
231 
232  solver.solve(this->system_matrix,
233  completely_distributed_solution,
234  this->system_rhs,
235  *this->ilu_preconditioner);
236 
237  if (verbose)
238  {
239  this->pcout << " -Iterative solver took: "
240  << solver_control.last_step()
241  << " steps to reach a residual norm of "
242  << solver_control.last_value() << std::endl;
243  }
244 
245  // Update constraints vector
246  constraints_used.distribute(completely_distributed_solution);
247 
248  // Update solution vectors
249  this->present_solution = completely_distributed_solution;
250  this->evaluation_point = this->present_solution;
251 }
