24 void
25 ParticleParticleContactForce<dim, contact_model, rolling_friction_model>::
26  calculate_particle_particle_contact_force(
27  typename dem_data_structures<dim>::adjacent_particle_pairs
28  &local_adjacent_particles,
29  typename dem_data_structures<dim>::adjacent_particle_pairs
30  &ghost_adjacent_particles,
31  typename dem_data_structures<dim>::adjacent_particle_pairs
32  &local_local_periodic_adjacent_particles,
33  typename dem_data_structures<dim>::adjacent_particle_pairs
34  &local_ghost_periodic_adjacent_particles,
35  typename dem_data_structures<dim>::adjacent_particle_pairs
36  &ghost_local_periodic_adjacent_particles,
37  const double dt,
38  std::vector<Tensor<1, 3>> &torque,
39  std::vector<Tensor<1, 3>> &force)
40 {
41  // Calculating the contact forces the local-local adjacent particles.
42  for (auto &&adjacent_particles_list :
43  local_adjacent_particles | boost::adaptors::map_values)
44  {
45  execute_contact_calculation<ContactType::local_particle_particle>(
46  adjacent_particles_list, torque, force, dt);
47  }
48 
49  // Calculating the contact forces the local-ghost adjacent particles.
50  for (auto &&adjacent_particles_list :
51  ghost_adjacent_particles | boost::adaptors::map_values)
52  {
53  execute_contact_calculation<ContactType::ghost_particle_particle>(
54  adjacent_particles_list, torque, force, dt);
55  }
56 
57  // Calculating the contact forces the local-local periodic adjacent particles.
58  for (auto &&periodic_adjacent_particles_list :
59  local_local_periodic_adjacent_particles | boost::adaptors::map_values)
60  {
61  execute_contact_calculation<
62  ContactType::local_periodic_particle_particle>(
63  periodic_adjacent_particles_list, torque, force, dt);
64  }
65 
66  // Calculating the contact forces the local-ghost periodic adjacent particles.
67  for (auto &&periodic_adjacent_particles_list :
68  local_ghost_periodic_adjacent_particles | boost::adaptors::map_values)
69  {
70  execute_contact_calculation<
71  ContactType::ghost_periodic_particle_particle>(
72  periodic_adjacent_particles_list, torque, force, dt);
73  }
74 
75  // Calculating the contact forces the ghost-local periodic adjacent particles.
76  for (auto &&periodic_adjacent_particles_list :
77  ghost_local_periodic_adjacent_particles | boost::adaptors::map_values)
78  {
79  execute_contact_calculation<
80  ContactType::ghost_local_periodic_particle_particle>(
81  periodic_adjacent_particles_list, torque, force, dt);
82  }
83 }
