149  virtual void
150  assemble_matrix_and_rhs()
151  {
152  if (some_particles_are_coupled)
153  {
154  if (this->get_current_newton_iteration() != 0)
155  {
156  force_on_ib();
157  }
158  integrate_particles();
159  update_precalculations_for_ib();
160  }
161  // Do the cut cell mapping only if it is the first Newton iteration or
162  // if the explicit position integration is false.
163  if (this->get_current_newton_iteration() == 0 ||
164  this->simulation_parameters.particlesParameters
165  ->explicit_position_integration_calculation == false)
166  {
167  if (all_spheres)
168  optimized_generate_cut_cells_map();
169  else
170  generate_cut_cells_map();
171  }
172  this->FluidDynamicsMatrixBased<dim>::assemble_system_matrix();
173 
174  sharp_edge();
175  }
