15 /**
16  * @brief Class that assembles the core of the Navier-Stokes equation with
17  * free surface using Cahn-Hilliard modeling.
18  * According to the following weak form:
19  * \f$ \nabla \cdot \mathbf{u} + \rho \mathbf{u} \cdot \nabla
20  * \mathbf{u} + \nabla p - \mu \nabla\cdot (\nabla \mathbf{u} +
21  * (\nabla \mathbf{u})^T)=0 \f$ with an SUPG and PSPG stabilization
22  *
23  * @tparam dim An integer that denotes the number of spatial dimensions
24  *
25  * @ingroup assemblers
26  */
27 template <int dim>
28 class GLSNavierStokesCahnHilliardAssemblerCore
29  : public NavierStokesAssemblerBase<dim>
30 {
31 public:
32  GLSNavierStokesCahnHilliardAssemblerCore(
33  const std::shared_ptr<SimulationControl> &simulation_control,
34  const SimulationParameters<dim> &nsparam)
35  : simulation_control(simulation_control)
36  , cahn_hilliard_parameters(nsparam.multiphysics.cahn_hilliard_parameters)
37  {}
