18  dem_parameters.model_parameters.particle_particle_contact_force_model;

87 set_rolling_resistance_model(

90  &particle_particle_contact_force_object)
91 {
92  RollingResistanceMethod rolling_resistance_method =
93  dem_parameters.model_parameters.rolling_resistance_method;
94 
95  switch (rolling_resistance_method)

99  particle_particle_contact_force_object =
100  std::make_shared<ParticleParticleContactForce<
101  dim,

103  RollingResistanceMethod::no_resistance>>(dem_parameters);
104  break;
105  }
106  case RollingResistanceMethod::constant_resistance:
107  {
108  particle_particle_contact_force_object =
109  std::make_shared<ParticleParticleContactForce<
110  dim,
111  particle_particle_contact_force_model,
112  RollingResistanceMethod::constant_resistance>>(dem_parameters);
113  break;
114  }
115  case RollingResistanceMethod::viscous_resistance:
116  {
117  particle_particle_contact_force_object =
118  std::make_shared<ParticleParticleContactForce<

121  RollingResistanceMethod::viscous_resistance>>(dem_parameters);
122  break;
123  }
124  default:
125  throw std::runtime_error("Invalid rolling resistance method");
126  }
