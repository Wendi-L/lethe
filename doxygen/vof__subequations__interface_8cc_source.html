37 std::shared_ptr<PhysicsScratchDataBase>
38 VOFSubequationsInterface<dim>::scratch_data_cast(
39  const VOFSubequationsID &subequation_id,
40  const FiniteElement<dim> &fe_subequation,
41  const Quadrature<dim> &quadrature,
42  const Mapping<dim> &mapping,
43  const FiniteElement<dim> &fe_input)
44 {
45  AssertThrow((std::find(this->active_subequations.begin(),
46  this->active_subequations.end(),
47  subequation_id) != this->active_subequations.end()),
48  ExcInternalError());
49 
50  if (subequation_id == VOFSubequationsID::phase_gradient_projection)
51  return std::make_shared<VOFPhaseGradientProjectionScratchData<dim>>(
52  fe_subequation, quadrature, mapping, fe_input);
53  else // At the moment, only one option is possible. This will change with the
54  // addition of other subequations to the interface.
55  return std::make_shared<VOFPhaseGradientProjectionScratchData<dim>>(
56  fe_subequation, quadrature, mapping, fe_input);
