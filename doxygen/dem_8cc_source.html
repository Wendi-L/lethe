55 template <int dim>
56 void
57 DEMSolver<dim>::setup_parameters()
58 {
59  // Print simulation starting information
60  pcout << std::endl;
61  std::stringstream ss;
62  ss << "Running on " << n_mpi_processes << " rank(s)";
63  announce_string(pcout, ss.str(), '*');
64 
65  // Check if the output directory exists
66  std::string output_dir_name = parameters.simulation_control.output_folder;
67  struct stat buffer;
68 
69  // If output directory does not exist, create it
70  if (this_mpi_process == 0)
71  {
72  if (stat(output_dir_name.c_str(), &buffer) != 0)
73  {
74  create_output_folder(output_dir_name);
75  }
76  }
77 
78  // Get the pointer of the only instance of the action manager
79  action_manager = DEMActionManager::get_action_manager();
80 
81  // Change the behavior of the timer for situations when you don't want outputs
82  if (parameters.timer.type == Parameters::Timer::Type::none)
83  computing_timer.disable_output();
84 
85  // Set the simulation control as transient DEM
86  simulation_control = std::make_shared<SimulationControlTransientDEM>(
87  parameters.simulation_control);
88 
89  // Setup load balancing parameters and attach the correct functions to the
90  // signals inside the triangulation
91  load_balancing.set_parameters(parameters.model_parameters);
92  load_balancing.copy_references(simulation_control,
93  triangulation,
94  particle_handler,
95  sparse_contacts_object);
96  load_balancing.connect_weight_signals();
97 
98  // Set the adaptive sparse contacts parameters
99  if (parameters.model_parameters.sparse_particle_contacts)
100  {
101  sparse_contacts_object.set_parameters(
102  parameters.model_parameters.granular_temperature_threshold,
103  parameters.model_parameters.solid_fraction_threshold,
104  parameters.model_parameters.advect_particles);
105  }
106 
107  // Set the distribution type and initialize the neighborhood threshold
108  setup_distribution_type();
109 
110  if (this_mpi_process == 0)
111  input_parameter_inspection(parameters,
112  pcout,
113  size_distribution_object_container);
114 
115  // Set the grid motion type
116  grid_motion_object =
117  std::make_shared<GridMotion<dim, dim>>(parameters.grid_motion,
118  simulation_control->get_time_step());
119 
120  // Set up the solid objects
121  setup_solid_objects();
122 
123  // Check if there are periodic boundaries
124  for (unsigned int i_bc = 0;
125  i_bc < parameters.boundary_conditions.bc_types.size();
126  ++i_bc)
127  {
128  if (parameters.boundary_conditions.bc_types[i_bc] ==
129  Parameters::Lagrangian::BCDEM::BoundaryType::periodic)
130  {
131  periodic_boundaries_object.set_periodic_boundaries_information(
132  parameters.boundary_conditions.periodic_boundary_0,
133  parameters.boundary_conditions.periodic_direction);
134  break;
135  }
136  }
137 
138  // Assign gravity/acceleration
139  g = parameters.lagrangian_physical_properties.g;
140 
141  // If this is a start simulation
142  if (parameters.restart.restart)
143  action_manager->restart_simulation();
144 }

146 template <int dim>
147 void
148 DEMSolver<dim>::setup_distribution_type()
149 {
150  // Use namespace and alias to make the code more readable
151  using namespace Parameters::Lagrangian;
152  LagrangianPhysicalProperties &lpp = parameters.lagrangian_physical_properties;
153 
154  maximum_particle_diameter = 0;
155  for (unsigned int particle_type = 0; particle_type < lpp.particle_type_number;
156  particle_type++)
157  {
158  switch (lpp.distribution_type.at(particle_type))
159  {
160  case SizeDistributionType::uniform:
161  size_distribution_object_container[particle_type] =
162  std::make_shared<UniformDistribution>(
163  lpp.particle_average_diameter.at(particle_type));
164  break;
165  case SizeDistributionType::normal:
166  size_distribution_object_container[particle_type] =
167  std::make_shared<NormalDistribution>(
168  lpp.particle_average_diameter.at(particle_type),
169  lpp.particle_size_std.at(particle_type),
170  lpp.seed_for_distributions[particle_type] + this_mpi_process);
171  break;
172  case SizeDistributionType::custom:
173  size_distribution_object_container[particle_type] =
174  std::make_shared<CustomDistribution>(
175  lpp.particle_custom_diameter.at(particle_type),
176  lpp.particle_custom_probability.at(particle_type),
177  lpp.seed_for_distributions[particle_type] + this_mpi_process);
178  break;
179  }
180 
181  maximum_particle_diameter = std::max(
182  maximum_particle_diameter,
183  size_distribution_object_container[particle_type]->find_max_diameter());
184  }
185 
186  neighborhood_threshold_squared =
187  std::pow(parameters.model_parameters.neighborhood_threshold *
188  maximum_particle_diameter,
189  2);
190 }

319 template <int dim>
320 void
321 DEMSolver<dim>::setup_triangulation_dependent_parameters()
322 {
323  // Find the smallest contact search frequency criterion between (smallest
324  // cell size - largest particle radius) and (security factor * (blob
325  // diameter - 1) * largest particle radius). This value is used in
326  // find_contact_detection_frequency function
327  smallest_contact_search_criterion =
328  std::min((GridTools::minimal_cell_diameter(triangulation) -
329  maximum_particle_diameter * 0.5),
330  (parameters.model_parameters.dynamic_contact_search_factor *
331  (parameters.model_parameters.neighborhood_threshold - 1) *
332  maximum_particle_diameter * 0.5));
333 
334  // Find the smallest cell size and use this as the floating mesh mapping
335  // criterion. The edge case comes when the cell are completely square/cubic.
336  // In that case, every sides of a cell are 2^-0.5 or 3^-0.5 times the
337  // cell_diameter. We want to refresh the mapping each time the solid-objet
338  // pass through a cell or there will be late contact detection. Thus, we use
339  // this value.
340  smallest_solid_object_mapping_criterion = [&] {
341  if constexpr (dim == 2) // 2^-0.5 * D_c,min
342  return 0.70710678118 * GridTools::minimal_cell_diameter(triangulation);
343  if constexpr (dim == 3) // 3^-0.5 * D_c,min
344  return 0.57735026919 * GridTools::minimal_cell_diameter(triangulation);
345  }();
346 
347  // Setup background dof
348  setup_background_dofs();
349 
350  // Set up the periodic boundaries (if PBC enabled)
351  periodic_boundaries_object.map_periodic_cells(
352  triangulation, periodic_boundaries_cells_information);
353 
354  // Set the periodic offset to contact managers and particles contact forces
355  // for periodic contact detection (if PBC enabled)
356  contact_manager.set_periodic_offset(
357  periodic_boundaries_object.get_periodic_offset_distance());
358  particle_particle_contact_force_object->set_periodic_offset(
359  periodic_boundaries_object.get_periodic_offset_distance());
360 
361  // Set up the local and ghost cells (if ASC enabled)
362  sparse_contacts_object.update_local_and_ghost_cell_set(background_dh);
363 }

365 template <int dim>
366 void
367 DEMSolver<dim>::setup_background_dofs()
368 {
369  FE_Q<dim> background_fe(1);
370  background_dh.distribute_dofs(background_fe);
371 
372  // Periodic nodes must be mapped otherwise the disabling of contacts will not
373  // propagate the mobility status to the periodic nodes during iterations.
374  // Identification of periodic nodes is done with the background constraints.
375  // Those constraints are not used for any matrix assembly in DEM solver, this
376  // approach comes from CFD-DEM coupling where void fraction constraints are
377  // used to achieve the periodic node mapping.
378  if (action_manager->check_periodic_boundaries_enabled() &&
379  action_manager->check_sparse_contacts_enabled())
380  {
381  IndexSet locally_relevant_dofs;
382  DoFTools::extract_locally_relevant_dofs(background_dh,
383  locally_relevant_dofs);
384 
385  background_constraints.clear();
386  background_constraints.reinit(locally_relevant_dofs);
387 
388  DoFTools::make_periodicity_constraints(
389  background_dh,
390  parameters.boundary_conditions.periodic_boundary_0,
391  parameters.boundary_conditions.periodic_boundary_1,
392  parameters.boundary_conditions.periodic_direction,
393  background_constraints);
394 
395  background_constraints.close();
396 
397  sparse_contacts_object.map_periodic_nodes(background_constraints);
398  }

401 template <int dim>
402 void
403 DEMSolver<dim>::load_balance()
404 {
405  load_balancing.check_load_balance_iteration();
406 
407  if (!action_manager->check_load_balance())
408  return;
409 
410  TimerOutput::Scope t(this->computing_timer, "Load balancing");
411  // Prepare particle handler for the adaptation of the triangulation to the
412  // load
413  particle_handler.prepare_for_coarsening_and_refinement();
414 
415  pcout << "-->Repartitionning triangulation" << std::endl;
416  triangulation.repartition();
417 
418  // Unpack the particle handler after the mesh has been repartitioned
419  particle_handler.unpack_after_coarsening_and_refinement();
420 
421  // If PBC are enabled, update the periodic cells
422  periodic_boundaries_object.map_periodic_cells(
423  triangulation, periodic_boundaries_cells_information);
424 
425  // If ASC is enabled, update the local and ghost cell set
426  sparse_contacts_object.update_local_and_ghost_cell_set(background_dh);
427 
428  // Update neighbors of cells after load balance
429  contact_manager.update_cell_neighbors(triangulation,
430  periodic_boundaries_cells_information);
431 
432  boundary_cell_object.build(
433  triangulation,
434  parameters.floating_walls,
435  parameters.boundary_conditions.outlet_boundaries,
436  parameters.mesh.check_for_diamond_cells,
437  parameters.mesh.expand_particle_wall_contact_search,
438  pcout);
439 
440  // Update the boundary information (if grid motion)
441  boundary_cell_object.update_boundary_info_after_grid_motion(
442  updated_boundary_points_and_normal_vectors);
443 
444  const auto average_minimum_maximum_cells =
445  Utilities::MPI::min_max_avg(triangulation.n_active_cells(),
446  mpi_communicator);
447 
448  const auto average_minimum_maximum_particles =
449  Utilities::MPI::min_max_avg(particle_handler.n_locally_owned_particles(),
450  mpi_communicator);
451 
452  pcout << "Load balance finished " << std::endl;
453  pcout
454  << "Average, minimum and maximum number of particles on the processors are "
455  << average_minimum_maximum_particles.avg << " , "
456  << average_minimum_maximum_particles.min << " and "
457  << average_minimum_maximum_particles.max << std::endl;
458  pcout << "Minimum and maximum number of cells owned by the processors are "
459  << average_minimum_maximum_cells.min << " and "
460  << average_minimum_maximum_cells.max << std::endl;
461 
462  setup_background_dofs();
463 }

646

648  const std::string particles_solution_name =

657  properties_class.get_properties_name());
658 
659  write_vtu_and_pvd<0, dim>(particles_pvdhandler,
660  particle_data_out,
661  folder,
662  particles_solution_name,
663  time,
664  iter,
665  group_files,
666  mpi_communicator);
667 
668  if (simulation_control->get_output_boundaries())
669  {
670  DataOutFaces<dim> data_out_faces;
671 
672  // Setup background dofs to initiate right sized boundary_id vector
673  Vector<float> boundary_id(background_dh.n_dofs());
674

800 template <int dim>
801 inline void
802 DEMSolver<dim>::sort_particles_into_subdomains_and_cells()
803 {
804  particle_handler.sort_particles_into_subdomains_and_cells();
805 
806  // Exchange ghost particles
807  particle_handler.exchange_ghost_particles(true);
808 
809  // Resize the displacement, force and torque containers only if the particles
810  // have changed subdomains
811  if (action_manager->check_resize_containers())
812  {
813  // Resize and reinitialize displacement container
814  displacement.resize(particle_handler.get_max_local_particle_index());
815  // Resize and reinitialize displacement container
816  force.resize(displacement.size());
817  torque.resize(displacement.size());
818  MOI.resize(displacement.size());
819 
820  // Updating moment of inertia container
821  for (auto &particle : particle_handler)
822  {
823  auto particle_properties = particle.get_properties();
824  MOI[particle.get_local_index()] =
825  0.1 * particle_properties[DEM::PropertiesIndex::mass] *
826  particle_properties[DEM::PropertiesIndex::dp] *
827  particle_properties[DEM::PropertiesIndex::dp];
828  }
829  }
830 
831  // Always reset the displacement values since we are doing a search detection
832  std::fill(displacement.begin(), displacement.end(), 0.);
833 }

835 template <int dim>
836 void
837 DEMSolver<dim>::solve()
838 {
839  // Set up the parameters
840  setup_parameters();
841 
842  // Reading mesh
843  read_mesh(parameters.mesh,
844  action_manager->check_restart_simulation(),
845  pcout,
846  triangulation,
847  parameters.boundary_conditions);
848 
849  // Set up functions and pointers according to parameters
850  setup_functions_and_pointers();
851 
852  // Read checkpoint if needed
853  read_checkpoint(computing_timer,
854  parameters,
855  simulation_control,
856  particles_pvdhandler,
857  grid_pvdhandler,
858  triangulation,
859  particle_handler,
860  insertion_object,
861  solid_surfaces);
862 
863  // Set up the various parameters that need the triangulation
864  setup_triangulation_dependent_parameters();
865 
866  // Build the mapping of the cell neighbors
867  contact_manager.execute_cell_neighbors_search(
868  triangulation, periodic_boundaries_cells_information);
869 
870  // Find boundary cells with faces
871  boundary_cell_object.build(
872  triangulation,
873  parameters.floating_walls,
874  parameters.boundary_conditions.outlet_boundaries,
875  parameters.mesh.check_for_diamond_cells,
876  parameters.mesh.expand_particle_wall_contact_search,
877  pcout);
878 
879  // DEM engine iterator
880  while (simulation_control->integrate())
881  {
882  simulation_control->print_progression(pcout);
883  if (simulation_control->is_verbose_iteration())
884  report_statistics();
885 
886  // Move grid and update the boundary information (if grid motion)
887  grid_motion_object->move_grid(triangulation);
888  boundary_cell_object.update_boundary_info_after_grid_motion(
889  updated_boundary_points_and_normal_vectors);
890 
891  // Insert particle if needed
892  insert_particles();
893 
894  // Load balancing (if load balancing enabled and if needed)
895  load_balance();
896 
897  // Check for contact search according to the contact detection method
898  contact_detection_iteration_check_function();
899 
900  // Check if solid object needs to be mapped with background mesh
901  // (if solid object)
902  find_floating_mesh_mapping_step(smallest_solid_object_mapping_criterion,
903  this->solid_surfaces);
904 
905  // Map solid objects if the action was triggered (if solid object)
906  if (action_manager->check_solid_object_search())
907  {
908  // Store information about floating mesh/background mesh
909  // intersection
910  for (unsigned int i_solid = 0; i_solid < solid_surfaces.size();
911  ++i_solid)
912  {
913  solid_surfaces_mesh_info[i_solid] =
914  solid_surfaces[i_solid]->map_solid_in_background_triangulation(
915  triangulation);
916  }
917 
918  for (unsigned int i_solid = 0; i_solid < solid_volumes.size();
919  ++i_solid)
920  {
921  solid_volumes_mesh_info[i_solid] =
922  solid_volumes[i_solid]->map_solid_in_background_triangulation(
923  triangulation);
924  }
925  }
926 
927  // Execute contact search if the action was triggered
928  if (action_manager->check_contact_search())
929  {
930  // Particles displacement if passing through a periodic boundary
931  // (if PBC enabled)
932  periodic_boundaries_object.execute_particles_displacement(
933  particle_handler, periodic_boundaries_cells_information);
934 
935  sort_particles_into_subdomains_and_cells();
936 
937  // Compute cell mobility (if ASC enabled)
938  sparse_contacts_object.identify_mobility_status(
939  background_dh,
940  particle_handler,
941  triangulation.n_active_cells(),
942  mpi_communicator);
943 
944  // Execute broad search by filling containers of particle-particle
945  // contact pair candidates and containers of particle-wall
946  // contact pair candidates
947  contact_manager.execute_particle_particle_broad_search(
948  particle_handler, sparse_contacts_object);
949 
950  contact_manager.execute_particle_wall_broad_search(
951  particle_handler,
952  boundary_cell_object,
953  solid_surfaces_mesh_info,
954  parameters.floating_walls,
955  simulation_control->get_current_time(),
956  sparse_contacts_object);
957 
958  // Update contacts, remove replicates and add new contact pairs
959  // to the contact containers when particles are exchanged between
960  // processors
961  contact_manager.update_contacts();
962 
963  // Updates the iterators to particles in local-local contact
964  // containers
965  contact_manager.update_local_particles_in_cells(particle_handler);
966 
967  // Execute fine search by updating particle-particle contact
968  // containers according to the neighborhood threshold
969  contact_manager.execute_particle_particle_fine_search(
970  neighborhood_threshold_squared);
971 
972  // Execute fine search by updating particle-wall contact
973  // containers according to the neighborhood threshold
974  contact_manager.execute_particle_wall_fine_search(
975  parameters.floating_walls,
976  simulation_control->get_current_time(),
977  neighborhood_threshold_squared);
978 
979  // Updating number of contact builds
980  contact_build_number++;
981  }
982  else
983  {
984  particle_handler.update_ghost_particles();
985  }
986 
987  // Particle-particle contact force
988  particle_particle_contact_force_object
989  ->calculate_particle_particle_contact_force(
990  contact_manager.get_local_adjacent_particles(),
991  contact_manager.get_ghost_adjacent_particles(),
992  contact_manager.get_local_local_periodic_adjacent_particles(),
993  contact_manager.get_local_ghost_periodic_adjacent_particles(),
994  contact_manager.get_ghost_local_periodic_adjacent_particles(),
995  simulation_control->get_time_step(),
996  torque,
997  force);
998 
999  // Update the boundary points and vectors (if grid motion)
1000  // We have to update the positions of the points on boundary faces and
1001  // their normal vectors here. The update_contacts deletes the
1002  // particle-wall contact candidate if it exists in the contact list.
1003  // As a result, when we update the points on boundary faces and their
1004  // normal vectors, update_contacts deletes it from the output of broad
1005  // search and they are not updated in the contact force calculations.
1006  grid_motion_object
1007  ->update_boundary_points_and_normal_vectors_in_contact_list(
1008  contact_manager.get_particle_wall_in_contact(),
1009  updated_boundary_points_and_normal_vectors);
1010 
1011  // Move solid objects (if solid object)
1012  move_solid_objects();
1013 
1014  // Particle-wall contact force
1015  particle_wall_contact_force();
1016 
1017  // Integration of force and velocity for new location of particles
1018  // The half step is calculated at the first iteration
1019  if (simulation_control->get_step_number() == 0)
1020  {
1021  integrator_object->integrate_half_step_location(
1022  particle_handler,
1023  g,
1024  simulation_control->get_time_step(),
1025  torque,
1026  force,
1027  MOI);
1028  }
1029  else
1030  {
1031  integrator_object->integrate(particle_handler,
1032  g,
1033  simulation_control->get_time_step(),
1034  torque,
1035  force,
1036  MOI,
1037  triangulation,
1038  sparse_contacts_object);
1039  }
1040 
1041  // Visualization
1042  if (simulation_control->is_output_iteration())
1043  write_output_results();
1044 
1045  // Calculation of forces and torques if needed
1046  if (parameters.forces_torques.calculate_force_torque)
1047  {
1048  if ((this_mpi_process == 0) &&
1049  (simulation_control->get_step_number() %
1050  parameters.forces_torques.output_frequency ==
1051  0) &&
1052  (parameters.forces_torques.force_torque_verbosity ==
1053  Parameters::Verbosity::verbose))
1054  {
1055  write_forces_torques_output_locally(
1056  forces_boundary_information[simulation_control
1057  ->get_step_number()],
1058  torques_boundary_information[simulation_control
1059  ->get_step_number()]);
1060  }
1061  }
1062 
1063  // Post-processing if needed
1064  post_process_results();
1065 
1066  // Write checkpoint if needed
1067  if (parameters.restart.checkpoint &&
1068  simulation_control->get_step_number() %
1069  parameters.restart.frequency ==
1070  0)
1071  {
1072  write_checkpoint(computing_timer,
1073  parameters,
1074  simulation_control,
1075  particles_pvdhandler,
1076  grid_pvdhandler,
1077  triangulation,
1078  particle_handler,
1079  insertion_object,
1080  solid_surfaces,
1081  pcout,
1082  mpi_communicator);
1083  }
1084 
1085  // Reset all trigger flags
1086  action_manager->reset_triggers();
1087  }
1088 
1089  finish_simulation();
1090 }
