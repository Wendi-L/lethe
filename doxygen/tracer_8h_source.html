5  * Implementation of tracer as an auxiliary physics.
6  * Equation solved:
7  * dT/dt + u * gradT = D * div(grad T) + f
8  * with T the tracer function, D the diffusivity and f the forcing
9  */

42 public:
43  Tracer(MultiphysicsInterface<dim> *multiphysics_interface,
44  const SimulationParameters<dim> &p_simulation_parameters,
45  std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
46  p_triangulation,
47  std::shared_ptr<SimulationControl> p_simulation_control)
48  : AuxiliaryPhysics<dim, GlobalVectorType>(
49  p_simulation_parameters.non_linear_solver.at(PhysicsID::tracer))
50  , multiphysics(multiphysics_interface)
51  , computing_timer(p_triangulation->get_communicator(),
52  this->pcout,
53  TimerOutput::summary,
54  TimerOutput::wall_times)
55  , simulation_parameters(p_simulation_parameters)
56  , triangulation(p_triangulation)
57  , simulation_control(p_simulation_control)
58  , dof_handler(*triangulation)
59  {
60  if (simulation_parameters.mesh.simplex)
61  {
62  // for simplex meshes
63  fe = std::make_shared<FE_SimplexP<dim>>(
64  simulation_parameters.fem_parameters.tracer_order);
65  mapping = std::make_shared<MappingFE<dim>>(*fe);
66  cell_quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
67  face_quadrature =
68  std::make_shared<QGaussSimplex<dim - 1>>(fe->degree + 1);
69  if (simulation_parameters.fem_parameters.tracer_uses_dg)
70  AssertThrow(
71  false,
72  ExcMessage(
73  "Discontinuous Galerkin elements cannot be simplex in the tracer physics."));
74  }
75  else
76  {
77  // Usual case, for quad/hex meshes
78  if (simulation_parameters.fem_parameters.tracer_uses_dg)
79  fe = std::make_shared<FE_DGQ<dim>>(
80  simulation_parameters.fem_parameters.tracer_order);
81  else
82  fe = std::make_shared<FE_Q<dim>>(
83  simulation_parameters.fem_parameters.tracer_order);
84  // The mapping must be a minimum of degree 1. This is necessary in the
85  // DG method since DG_Q(0) elements are adequate but a mapping of degree
86  // 0 does not make sense.
87  mapping = std::make_shared<MappingQ<dim>>(std::max(int(fe->degree), 1));
88  cell_quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
89  face_quadrature = std::make_shared<QGauss<dim - 1>>(fe->degree + 1);
90  }
91 
92  // Allocate solution transfer
93  solution_transfer = std::make_shared<
94  parallel::distributed::SolutionTransfer<dim, GlobalVectorType>>(
95  dof_handler);
96 
97  // Set size of previous solutions using BDF schemes information
98  previous_solutions.resize(maximum_number_of_previous_solutions());
99 
100  // Prepare previous solutions transfer
101  previous_solutions_transfer.reserve(previous_solutions.size());
102  for (unsigned int i = 0; i < previous_solutions.size(); ++i)
103  {
104  previous_solutions_transfer.emplace_back(
105  parallel::distributed::SolutionTransfer<dim, GlobalVectorType>(
106  this->dof_handler));
107  }
108 
109  // Change the behavior of the timer for situations when you don't want
110  // outputs
111  if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
112  this->computing_timer.disable_output();
113  }

428  /**
429  * @brief Get the Lethe boundary indicator for a given triangulation boundary while
430  * carrying the appropriate checks.
431  *
432  * @param[in] triangulation_boundary_id The boundary id of the triangulation
433  *
434  * NOTE: This function is a temporary function. It will be deprecated in a
435  * future PR once boundary conditions have been refactored using maps.
436  */
437  unsigned int
438  get_lethe_boundary_index(const types::boundary_id &triangulation_boundary_id)
439  {
440  // Identify which boundary condition corresponds to the boundary id. If
441  // this boundary condition is not identified, then exit the simulation
442  // instead of assuming an outlet.
443  const auto lethe_boundary_id = std::find(
444  this->simulation_parameters.boundary_conditions_tracer.id.begin(),
445  this->simulation_parameters.boundary_conditions_tracer.id.end(),
446  triangulation_boundary_id);
447 
448  AssertThrow(
449  lethe_boundary_id !=
450  this->simulation_parameters.boundary_conditions_tracer.id.end(),
451  ExcMessage("The boundary condition with id " +
452  std::to_string(triangulation_boundary_id) +
453  " is not present in the tracer boundary conditions."));
454 
455  return (lethe_boundary_id -
456  this->simulation_parameters.boundary_conditions_tracer.id.begin());
457  }

459  /**
460  * @brief Helper function to reinit the face velocity with the adequate solution.
461  * This prevents code duplication throughout the tracer. The function looks at
462  * the multiphysics interface to decide if the velocity is a block velocity or
463  * a regular velocity. Furthermore, it also checks if a time-averaged solution
464  * is required. Otherwise the code here would be copied four times.
465  *
466  * @param[in] velocity_cell the iterator to the cell where the velocity is to
467  * be reinitialized.
468  *
469  * @param[in] face_no the number of the face where the velocity is to be
470  * reinitialized.
471  *
472  * @param[in,out] scratch_data the scratch data to be used for the
473  * reinitialization.
474  */
475  inline void
476  reinit_face_velocity_with_adequate_solution(
477  const typename DoFHandler<dim>::active_cell_iterator &velocity_cell,
478  const unsigned int &face_no,
479  TracerScratchData<dim> &scratch_data)
480  {
481  if (multiphysics->fluid_dynamics_is_block())
482  {
483  // Check if the post processed variable needs to be calculated with the
484  // average velocity profile or the fluid solution.
485  if (this->simulation_parameters.initial_condition->type ==
486  Parameters::InitialConditionType::average_velocity_profile &&
487  !this->simulation_parameters.multiphysics.fluid_dynamics &&
488  simulation_control->get_current_time() >
489  this->simulation_parameters.post_processing.initial_time)
490  {
491  scratch_data.reinit_face_velocity(
492  velocity_cell,
493  face_no,
494  *multiphysics->get_block_time_average_solution(
495  PhysicsID::fluid_dynamics),
496  this->simulation_parameters.ale,
497  this->simulation_parameters.tracer_drift_velocity.drift_velocity);
498  }
499  else
500  {
501  scratch_data.reinit_face_velocity(
502  velocity_cell,
503  face_no,
504  *multiphysics->get_block_solution(PhysicsID::fluid_dynamics),
505  this->simulation_parameters.ale,
506  this->simulation_parameters.tracer_drift_velocity.drift_velocity);
507  }
508  }
509  else
510  {
511  // Check if the post processed variable needs to be calculated with the
512  // average velocity profile or the fluid solution.
513  if (this->simulation_parameters.initial_condition->type ==
514  Parameters::InitialConditionType::average_velocity_profile &&
515  !this->simulation_parameters.multiphysics.fluid_dynamics &&
516  simulation_control->get_current_time() >
517  this->simulation_parameters.post_processing.initial_time)
518  {
519  scratch_data.reinit_face_velocity(
520  velocity_cell,
521  face_no,
522  *multiphysics->get_time_average_solution(
523  PhysicsID::fluid_dynamics),
524  this->simulation_parameters.ale,
525  this->simulation_parameters.tracer_drift_velocity.drift_velocity);
526  }
527  else
528  {
529  scratch_data.reinit_face_velocity(
530  velocity_cell,
531  face_no,
532  *multiphysics->get_solution(PhysicsID::fluid_dynamics),
533  this->simulation_parameters.ale,
534  this->simulation_parameters.tracer_drift_velocity.drift_velocity);
535  }
536  }
537  }
