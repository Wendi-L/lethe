38 template <int dim,
39  ParticleParticleContactForceModel contact_model,
40  RollingResistanceMethod rolling_friction_model>
41 void
42 ParticlesForceChains<dim, contact_model, rolling_friction_model>::
43  multi_general_cell(Triangulation<1, 3> &tria,
44  const std::vector<Point<3>> &vertices)
45 {
46  const unsigned int n_cells = vertices.size() / 2;
47  std::vector<CellData<1>> cells(n_cells, CellData<1>());
48  for (unsigned int i = 0; i < n_cells; ++i)
49  {
50  cells[i].vertices[0] = 2 * i;
51  cells[i].vertices[1] = 2 * i + 1;
52  cells[i].material_id = 0;
53  };
54  tria.create_triangulation(vertices, cells, SubCellData());
55 }

82 }

84 template <int dim,
85  ParticleParticleContactForceModel contact_model,
86  RollingResistanceMethod rolling_friction_model>
87 void
88 ParticlesForceChains<dim, contact_model, rolling_friction_model>::
89  write_force_chains(const DEMSolverParameters<dim> &dem_parameters,
90  PVDHandler &pvd_handler,
91  MPI_Comm mpi_communicator,
92  const std::string folder,
93  const unsigned int iter,
94  const double time)
95 {
96  Triangulation<1, 3> triangulation;
97  multi_general_cell(triangulation, vertices);
98  DoFHandler<1, 3> force_dh(triangulation);
99  DataOut<1, 3> data_out;
100  data_out.attach_dof_handler(force_dh);
101 
102  Vector<float> force_values(triangulation.n_active_cells());
103  for (unsigned int i = 0; i < force_values.size(); ++i)
104  {
105  force_values[i] = force_normal[i];
106  }
107  data_out.add_data_vector(force_values, "force");
108  data_out.build_patches();
109 
110  const std::string file_prefix =
111  dem_parameters.simulation_control.output_name + "-force_chains";
112  const std::string face_filename =
113  (folder + file_prefix + "." + Utilities::int_to_string(iter, 5) + ".vtu");
114  data_out.write_vtu_in_parallel(face_filename.c_str(), mpi_communicator);
115 
116  std::vector<std::string> filenames;
117  filenames.push_back(file_prefix + "." + Utilities::int_to_string(iter, 5) +
118  ".vtu");
119 
120  std::string pvtu_filename =
121  (file_prefix + "." + Utilities::int_to_string(iter, 5) + ".pvtu");
122 
123  std::string pvtu_filename_with_folder = folder + pvtu_filename;
124  std::ofstream master_output(pvtu_filename_with_folder.c_str());
125 
126  data_out.write_pvtu_record(master_output, filenames);
127 
128  std::string pvdPrefix = (folder + file_prefix + ".pvd");
129  pvd_handler.append(time, pvtu_filename);
130  std::ofstream pvd_output(pvdPrefix.c_str());
131  DataOutBase::write_pvd_record(pvd_output, pvd_handler.times_and_names);
132 }
