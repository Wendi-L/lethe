26 /**
27  * @brief Store the information required by the assembly procedure
28  * for a VOF free surface equation. Consequently, this class
29  * calculates the phase values (values, gradients, laplacians) and the shape
30  * method (values, gradients, laplacians) at all the gauss points for all
31  * degrees of freedom and stores it into arrays.
32  * This class serves as a separation between the evaluation at the gauss point
33  * of the variables of interest and their use in the assembly, which is carried
34  * out by the assembler methods.
35  *
36  * @tparam dim An integer that denotes the dimension of the space in which
37  * the flow is solved
38  *
39  * @ingroup solvers
40  **/

120  /** @brief Reinitialize the content of the scratch
121  *
122  * Using the FeValues and the content of the solutions and previous solutions,
123  * fills all of the class member of the scratch
124  *
125  * @tparam VectorType The Vector type used for the solvers
126  *
127  * @param[in] cell The cell over which the assembly is being carried.
128  * This cell must be compatible with the FE which is used to fill the
129  * FeValues.
130  *
131  * @param[in] current_solution The present value of the solution for the VOF.
132  *
133  * @param[in] previous_solutions The solutions at the previous time steps.

137  template <typename VectorType>
138  void
139  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
140  const VectorType &current_solution,
141  const std::vector<VectorType> &previous_solutions)
142  {
143  fe_values_vof.reinit(cell);
144  this->quadrature_points = fe_values_vof.get_quadrature_points();
145  auto &fe_vof = fe_values_vof.get_fe();
146 
147  // Compute cell diameter
148  double cell_measure =
149  compute_cell_measure_with_JxW(this->fe_values_vof.get_JxW_values());
150  this->cell_size = compute_cell_diameter<dim>(cell_measure, fe_vof.degree);
151 
152  fe_values_vof.get_function_values(current_solution,
153  this->present_phase_values);
154  fe_values_vof.get_function_gradients(current_solution,
155  this->phase_gradients);
156  fe_values_vof.get_function_laplacians(current_solution,
157  this->phase_laplacians);
158  fe_values_vof.get_function_gradients(previous_solutions[0],
159  this->previous_phase_gradients);
160 
161 
162  // Gather previous vof values
163  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
164  {
165  fe_values_vof.get_function_values(previous_solutions[p],
166  this->previous_phase_values[p]);
167  }
168 
169  for (unsigned int q = 0; q < this->n_q_points; ++q)
170  {
171  this->JxW[q] = fe_values_vof.JxW(q);
172 
173  for (unsigned int k = 0; k < this->n_dofs; ++k)
174  {
175  // Shape function
176  this->phi[q][k] = fe_values_vof.shape_value(k, q);
177  this->grad_phi[q][k] = fe_values_vof.shape_grad(k, q);
178  this->hess_phi[q][k] = fe_values_vof.shape_hessian(k, q);
179  this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
180  }
181  }
182  }

184  /** @brief Reinitialize the velocity, calculated by the Fluid Dynamics
185  *
186  * @tparam VectorType The Vector type used for the solvers
187  *
188  * @param cell The cell for which the velocity is reinitialized
189  * This cell must be compatible with the Fluid Dynamics FE
190  *
191  * @param current_solution The present value of the solution for \f$[u,p]\f$
192  *
193  * @param previous_solutions Vector of \f$n\f$ @p VectorType containers of
194  * previous fluid dynamic solutions (\f$[u,p]\f$). \f$n\f$ depends on the BDF
195  * scheme selected for time-stepping.
196  *
197  * @param ale ALE parameters
198  *
199  */
200 
201  template <typename VectorType>
202  void
203  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
204  const VectorType &current_solution,
205  const std::vector<VectorType> &previous_solutions,
206  const Parameters::ALE<dim> &ale)
207  {
208  fe_values_fd.reinit(cell);
209 
210  fe_values_fd[velocities_fd].get_function_values(current_solution,
211  velocity_values);
212  fe_values_fd[velocities_fd].get_function_gradients(
213  current_solution, velocity_gradient_values);
214 
215  for (unsigned int q = 0; q < this->n_q_points; ++q)
216  {
217  this->velocity_divergences[q] =
218  trace(this->velocity_gradient_values[q]);
219  }
220 
221  // Gather previous velocity values
222  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
223  {
224  fe_values_fd[velocities_fd].get_function_values(
225  previous_solutions[p], this->previous_velocity_values[p]);
226  }
227 
228  if (ale.enabled())
229  {
230  // ALE enabled, so extract the ALE velocity and subtract it from the
231  // velocity obtained from the fluid dynamics
232  Tensor<1, dim> velocity_ale;
233  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
234  ale.velocity;
235  Vector<double> velocity_ale_vector(dim);
236 
237  for (unsigned int q = 0; q < n_q_points; ++q)
238  {
239  velocity_ale_function->vector_value(quadrature_points[q],
240  velocity_ale_vector);
241  for (unsigned int d = 0; d < dim; ++d)
242  velocity_ale[d] = velocity_ale_vector[d];
243 
244  velocity_values[q] -= velocity_ale;
245 
246  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
247  {
248  this->previous_velocity_values[p][q] -= velocity_ale;
249  }
250  }
251  }
252 
253  // Extrapolate velocity to t+dt using the BDF scheme if the simulation is
254  // transient
255  const auto method = this->simulation_control->get_assembly_method();
256  if (is_bdf(method))
257  {
258  // Extrapolate velocity
259  std::vector<double> time_vector =
260  this->simulation_control->get_simulation_times();
261  bdf_extrapolate(time_vector,
262  this->previous_velocity_values,
263  number_of_previous_solutions(method),
264  this->velocity_values);
265  }
266  }

398  /**
399  * @brief Reinitialize the content of the object for a given cell.
400  *
401  * @tparam VectorType Vector type used for the solvers.
402  *
403  * @param[in] cell Cell over which the reinitialization is carried out.
404  *
405  * @param[in] current_solution Present values of the solution for the
406  * projected phase fraction gradient.
407  */
408  template <typename VectorType>
409  void
410  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
411  const VectorType &current_solution)
412  {
413  // Reinitialize FEValues
414  this->fe_values_phase_gradient_projection.reinit(cell);
415  auto &fe_phase_gradient_projection =
416  this->fe_values_phase_gradient_projection.get_fe();
417 
418  // Get cell measure
419  this->JxW = this->fe_values_phase_gradient_projection.get_JxW_values();
420  double cell_measure = compute_cell_measure_with_JxW(this->JxW);
421  this->cell_size =
422  compute_cell_diameter<dim>(cell_measure,
423  fe_phase_gradient_projection.degree);
424 
425  // Gather present solutions
426  this->fe_values_phase_gradient_projection[this->phase_fraction_gradients]
427  .get_function_values(current_solution,
428  this->present_phase_gradient_projection_values);
429  this->fe_values_phase_gradient_projection[this->phase_fraction_gradients]
430  .get_function_gradients(
431  current_solution, this->present_phase_gradient_projection_gradients);
432 
433  // Shape functions
434  for (unsigned int q = 0; q < this->n_q_points; ++q)
435  {
436  for (unsigned int k = 0; k < this->n_dofs; ++k)
437  {
438  this->phi[q][k] = this
439  ->fe_values_phase_gradient_projection
440  [this->phase_fraction_gradients]
441  .value(k, q);
442  this->grad_phi[q][k] = this
443  ->fe_values_phase_gradient_projection
444  [this->phase_fraction_gradients]
445  .gradient(k, q);
446  }
447  }
448  }
