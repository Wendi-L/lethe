6double
7calculate_corrected_contact_radius(const double effective_radius,
8 const double effective_youngs_modulus,
9 const double effective_real_youngs_modulus,
10 const double normal_force_norm)
11{
12 const double contact_radius = pow((3 * normal_force_norm * effective_radius) /
13 (4 * effective_youngs_modulus),
14 (1.0 / 3.0));
15
16 // In certain cases, the simulation effective Young’s modulus
17 // can be chosen a few orders lower than the real effective Young’s modulus
18 // to ensure simulation stability (related to Rayleigh's time).
19 // As a consequence, the contact radius is overestimated, so a correctional
20 // factor is applied here.
21 return contact_radius *
22 pow(effective_youngs_modulus / effective_real_youngs_modulus,
23 1.0 / 5.0);
24}

27calculate_macrocontact_resistance(const double harmonic_particle_conductivity,
28 const double contact_radius)

30 return 0.5 / (contact_radius * harmonic_particle_conductivity + DBL_MIN);

33double
34calculate_microcontact_resistance(const double equivalent_surface_slope,
35 const double equivalent_surface_roughness,
36 const double effective_microhardness,
37 const double contact_radius_squared,
38 const double harmonic_particle_conductivity,
39 const double maximum_pressure)
40{
41 // 1.184 = (1+0.96/2)/1.25
42 return 1.184 /
43 (M_PI * harmonic_particle_conductivity * contact_radius_squared) *
44 (equivalent_surface_roughness / equivalent_surface_slope) *
45 pow(effective_microhardness / (maximum_pressure + DBL_MIN), 0.96);
46
47 // Van Lew, J. T. (2016). On thermal characterization of breeder pebble beds
48 // with microscale numerical modeling of thermofluid and pebble-pebble
49 // interactions (Doctoral dissertation, University of California, Los Angeles)
50 // Equations 3.25 to 3.27.
51}

53double
54calculate_solid_macrogap_resistance(const double radius_one,
55 const double radius_two,
56 const double thermal_conductivity_one,
57 const double thermal_conductivity_two,
58 const double contact_radius_squared)
59{
60 // resistance_i = characteristic length parallel to heat flux /
61 // (thermal_conductivity
62 // * characteristic area perpendicular to the heat flux)
63 const double resistance_one =
64 0.25 * M_PI * radius_one /
65 (M_PI * (radius_one * radius_one - contact_radius_squared) *
66 thermal_conductivity_one);
67 const double resistance_two =
68 0.25 * M_PI * radius_two /
69 (M_PI * (radius_two * radius_two - contact_radius_squared) *
70 thermal_conductivity_two);
71
72 return resistance_one + resistance_two;
73}

75double
76calculate_interstitial_gas_microgap_resistance(
77 const double equivalent_surface_roughness,
78 const double contact_radius_squared,
79 const double gas_parameter_m,
80 const double thermal_conductivity_gas,
81 const double maximum_pressure,
82 const double effective_microhardness)
83{
84 const double x_1 = 2.0 * maximum_pressure / effective_microhardness;
85 const double x_2 = 0.03 * maximum_pressure / effective_microhardness;
86
87 // If the values are out of the valid range ]0, 2[ of the erfc^-1 function,
88 // this resistance has no physical meaning. Particles are too close or not
89 // close enough for a microgap between them to make sense. This resistance is
90 // used in parallel with another one, so putting it to infinity ignores it in
91 // the total resistance.
92 if (x_1 >= 2.0 || x_1 <= 0.0)
93 {
94 return INFINITY;
95 }
96
97 const double a_1 = boost::math::erfc_inv(x_1);
98 const double a_2 = boost::math::erfc_inv(x_2) - a_1;
99
100 // 2.82842712475 = 2*sqrt(2)
101 return (2.82842712475 * equivalent_surface_roughness * a_2) /
102 (M_PI * thermal_conductivity_gas * contact_radius_squared *
103 std::log(1 + a_2 / (a_1 + gas_parameter_m /
104 (2.82842712475 *
105 equivalent_surface_roughness))));
106}

108double
109calculate_interstitial_gas_macrogap_resistance(
110 const double harmonic_radius,
111 const double thermal_conductivity_gas,
112 const double contact_radius_squared,
113 const double gas_parameter_m)
114{
115 const double A =
116 2. * sqrt(harmonic_radius * harmonic_radius - contact_radius_squared);
117 const double S = 2. * harmonic_radius -
118 contact_radius_squared / harmonic_radius + gas_parameter_m;
119
120 return 2.0 / (M_PI * thermal_conductivity_gas * (S * log(S / (S - A)) - A));
121}

123void
124calculate_contact_thermal_conductance(
125 const double radius_one,
126 const double radius_two,
127 const double effective_youngs_modulus,
128 const double effective_real_youngs_modulus,
129 const double equivalent_surface_roughness,
130 const double equivalent_surface_slope,
131 const double effective_microhardness,
132 const double thermal_conductivity_one,
133 const double thermal_conductivity_two,
134 const double thermal_conductivity_gas,
135 const double gas_parameter_m,
136 const double normal_overlap,
137 const double normal_force_norm,
138 double &thermal_conductance)
139{
140 const double harmonic_particle_conductivity =
141 harmonic_mean(thermal_conductivity_one, thermal_conductivity_two);
142 const double harmonic_radius = harmonic_mean(radius_one, radius_two);
143
144 // Calculation of contact radius
145 const double contact_radius =
146 calculate_corrected_contact_radius(harmonic_radius * 0.5,
147 effective_youngs_modulus,
148 effective_real_youngs_modulus,
149 normal_force_norm);
150
151 // Squared contact radius, often used in resistances calculations.
152 const double contact_radius_squared = contact_radius * contact_radius;
153
154 // In the same way as the contact radius, if the effective Young's modulus is
155 // underestimated, the normal overlap is overestimated so a correctional
156 // factor is applied here.
157 const double corrected_normal_overlap =
158 normal_overlap *
159 pow(effective_youngs_modulus / effective_real_youngs_modulus, 2.0 / 3.0);
160 const double maximum_pressure =
161 (2.0 * effective_real_youngs_modulus * corrected_normal_overlap) /
162 (M_PI * contact_radius + DBL_MIN);
163
164 // Calculation of each thermal resistance
165 const double resistance_macrocontact =
166 calculate_macrocontact_resistance(harmonic_particle_conductivity,
167 contact_radius);
168
169 const double resistance_microcontact =
170 calculate_microcontact_resistance(equivalent_surface_slope,
171 equivalent_surface_roughness,
172 effective_microhardness,
173 contact_radius_squared,
174 harmonic_particle_conductivity,
175 maximum_pressure);
176
177 const double resistance_solid_macrogap =
178 calculate_solid_macrogap_resistance(radius_one,
179 radius_two,
180 thermal_conductivity_one,
181 thermal_conductivity_two,
182 contact_radius_squared);
183
184 const double resistance_gas_microgap =
185 calculate_interstitial_gas_microgap_resistance(equivalent_surface_roughness,
186 contact_radius_squared,
187 gas_parameter_m,
188 thermal_conductivity_gas,
189 maximum_pressure,
190 effective_microhardness);
191
192 const double resistance_gas_macrogap =
193 calculate_interstitial_gas_macrogap_resistance(harmonic_radius,
194 thermal_conductivity_gas,
195 contact_radius_squared,
196 gas_parameter_m);
197
198 // Calculation of the final thermal conductance (1 / total resistance)
199 // conductance = 1 / resistance contact area pathway + 1 / resistance magrogap
200 // pathway)
201 thermal_conductance =
202 1.0 / (resistance_macrocontact + 1.0 / (1.0 / resistance_microcontact +
203 1.0 / resistance_gas_microgap)) +
204 1.0 / (resistance_solid_macrogap + resistance_gas_macrogap);
205}

207void
208apply_heat_transfer_on_local_particles(const double temperature_one,
209 const double temperature_two,
210 const double thermal_conductance,
211 double &particle_one_heat_transfer_rate,
212 double &particle_two_heat_transfer_rate)
213{
214 const double heat_transfer_rate =
215 thermal_conductance * (temperature_two - temperature_one);
216 particle_one_heat_transfer_rate += heat_transfer_rate;
217 particle_two_heat_transfer_rate -= heat_transfer_rate;
218}
