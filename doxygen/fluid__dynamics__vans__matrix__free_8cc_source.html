20template <int dim>
21FluidDynamicsVANSMatrixFree<dim>::FluidDynamicsVANSMatrixFree(
22 CFDDEMSimulationParameters<dim> &param)
23 : FluidDynamicsMatrixFree<dim>(param.cfd_parameters)
24 , cfd_dem_simulation_parameters(param)
25 , particle_mapping(1)
26 , particle_handler(*this->triangulation,
27 particle_mapping,
28 DEM::CFDDEMProperties::n_properties)
29 , void_fraction_manager(
30 &(*this->triangulation),
31 param.void_fraction,
32 this->cfd_dem_simulation_parameters.cfd_parameters.linear_solver.at(
33 PhysicsID::fluid_dynamics),
34 &particle_handler,
35 this->cfd_dem_simulation_parameters.cfd_parameters.fem_parameters
36 .void_fraction_order,
37 this->cfd_dem_simulation_parameters.cfd_parameters.mesh.simplex,
38 this->pcout)
39 , has_periodic_boundaries(false)
40{
41 unsigned int n_pbc = 0;
42 for (auto const &[id, type] :
43 cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions.type)
44 {
45 if (type == BoundaryConditions::BoundaryType::periodic)
46 {
47 if (n_pbc++ > 1)
48 {
49 throw std::runtime_error(
50 "GLS VANS solver does not support more than one periodic boundary condition.");
51 }
52 else
53 {
54 has_periodic_boundaries = true;
55 }
56 }
57 }
58
59 // The default MatrixFree solver sets a system_operator. We override the
60 // Navier-Stokes operator with the volume-averaged Navier-Stokes operator.
61 this->system_operator = std::make_shared<VANSOperator<dim, double>>(
62 cfd_dem_simulation_parameters.cfd_dem);
63}

96template <int dim>
97void
98FluidDynamicsVANSMatrixFree<dim>::solve()
99{
100 this->computing_timer.enter_subsection("Read mesh and manifolds");
101
102 read_mesh_and_manifolds(
103 *this->triangulation,
104 this->simulation_parameters.mesh,
105 this->simulation_parameters.manifolds_parameters,
106 this->simulation_parameters.restart_parameters.restart,
107 this->simulation_parameters.boundary_conditions);
108
109 this->computing_timer.leave_subsection("Read mesh and manifolds");
110
111 this->setup_dofs();
112 this->set_initial_condition(
113 this->simulation_parameters.initial_condition->type,
114 this->simulation_parameters.restart_parameters.restart);
115
116 // Only needed if other physics apart from fluid dynamics are enabled.
117 if (this->multiphysics->get_active_physics().size() > 1)
118 this->update_multiphysics_time_average_solution();
119
120 while (this->simulation_control->integrate())
121 {
122 if (this->forcing_function)
123 this->forcing_function->set_time(
124 this->simulation_control->get_current_time());
125
126 this->update_boundary_conditions();
127 this->multiphysics->update_boundary_conditions();
128
129 this->simulation_control->print_progression(this->pcout);
130 this->dynamic_flow_control();
131
132 if (!this->simulation_control->is_at_start())
133 {
134 this->refine_mesh();
135 }
136
137 if (is_bdf(this->simulation_control->get_assembly_method()))
138 {
139 this->computing_timer.enter_subsection(
140 "Calculate time derivative previous solutions");
141
142 this->calculate_time_derivative_previous_solutions();
143 this->time_derivative_previous_solutions.update_ghost_values();
144 this->system_operator->evaluate_time_derivative_previous_solutions(
145 this->time_derivative_previous_solutions);
146
147 this->computing_timer.leave_subsection(
148 "Calculate time derivative previous solutions");
149
150 if (this->simulation_parameters.flow_control.enable_flow_control)
151 this->system_operator->update_beta_force(
152 this->flow_control.get_beta());
153 }
154
155 // Calculate the void fraction and evaluate it within the matrix-free
156 // operator
157 {
158 TimerOutput::Scope t(this->computing_timer, "Calculate void fraction");
159 void_fraction_manager.calculate_void_fraction(
160 this->simulation_control->get_current_time());
161
162 // The base matrix-free operator is not aware of the void fraction. We
163 // must do a cast here to ensure that the operator is of the right type
164 if (auto mf_operator = dynamic_cast<VANSOperator<dim, double> *>(
165 this->system_operator.get()))
166 mf_operator->evaluate_void_fraction(void_fraction_manager);
167 }
168
169 this->iterate();
170 this->postprocess(false);
171 this->finish_time_step();
172
173 if (this->simulation_parameters.timer.type ==
174 Parameters::Timer::Type::iteration)
175 this->print_mg_setup_times();
176 }
177
178 if (this->simulation_parameters.timer.type == Parameters::Timer::Type::end)
179 this->print_mg_setup_times();
180
181 this->finish_simulation();
182}
