55 present_solution,
56 time_derivative_previous_solutions);

59template <int dim>
60FluidDynamicsVANSMatrixFree<dim>::FluidDynamicsVANSMatrixFree(
61 CFDDEMSimulationParameters<dim> &param)
62 : FluidDynamicsMatrixFree<dim>(param.cfd_parameters)
63 , cfd_dem_simulation_parameters(param)
64 , particle_mapping(1)
65 , particle_handler(*this->triangulation,
66 particle_mapping,
67 DEM::CFDDEMProperties::n_properties)
68 , void_fraction_manager(
69 &(*this->triangulation),
70 param.void_fraction,
71 this->cfd_dem_simulation_parameters.cfd_parameters.linear_solver.at(
72 PhysicsID::fluid_dynamics),
73 &particle_handler,
74 this->cfd_dem_simulation_parameters.cfd_parameters.fem_parameters
75 .void_fraction_order,
76 this->cfd_dem_simulation_parameters.cfd_parameters.mesh.simplex,
77 this->pcout)
78 , has_periodic_boundaries(false)
79{
80 unsigned int n_pbc = 0;
81 for (auto const &[id, type] :
82 cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions.type)
83 {
84 if (type == BoundaryConditions::BoundaryType::periodic)
85 {
86 if (n_pbc++ > 1)
87 {
88 throw std::runtime_error(
89 "GLS VANS solver does not support more than one periodic boundary condition.");
90 }
91 else
92 {
93 has_periodic_boundaries = true;
94 }
95 }
96 }
97
98 // The default MatrixFree solver sets a system_operator. We override the
99 // Navier-Stokes operator with the volume-averaged Navier-Stokes operator.
100 this->system_operator = std::make_shared<VANSOperator<dim, double>>(
101 cfd_dem_simulation_parameters.cfd_dem);
102}

135template <int dim>
136void
137FluidDynamicsVANSMatrixFree<dim>::solve()
138{
139 this->computing_timer.enter_subsection("Read mesh and manifolds");
140
141 read_mesh_and_manifolds(
142 *this->triangulation,
143 this->simulation_parameters.mesh,
144 this->simulation_parameters.manifolds_parameters,
145 this->simulation_parameters.restart_parameters.restart,
146 this->simulation_parameters.boundary_conditions);
147
148 this->computing_timer.leave_subsection("Read mesh and manifolds");
149
150 this->setup_dofs();
151 this->set_initial_condition(
152 this->simulation_parameters.initial_condition->type,
153 this->simulation_parameters.restart_parameters.restart);
154
155 // Only needed if other physics apart from fluid dynamics are enabled.
156 if (this->multiphysics->get_active_physics().size() > 1)
157 this->update_multiphysics_time_average_solution();
158
159 while (this->simulation_control->integrate())
160 {
161 if (this->forcing_function)
162 this->forcing_function->set_time(
163 this->simulation_control->get_current_time());
164
165 this->update_boundary_conditions();
166 this->multiphysics->update_boundary_conditions();
167
168 this->simulation_control->print_progression(this->pcout);
169 this->dynamic_flow_control();
170
171 if (!this->simulation_control->is_at_start())
172 {
173 this->refine_mesh();
174 }
175
176 if (is_bdf(this->simulation_control->get_assembly_method()))
177 {
178 this->computing_timer.enter_subsection(
179 "Calculate time derivative previous solutions");
180
181 this->calculate_time_derivative_previous_solutions();
182 this->time_derivative_previous_solutions.update_ghost_values();
183 this->system_operator->evaluate_time_derivative_previous_solutions(
184 this->time_derivative_previous_solutions);
185
186 this->computing_timer.leave_subsection(
187 "Calculate time derivative previous solutions");
188
189 if (this->simulation_parameters.flow_control.enable_flow_control)
190 this->system_operator->update_beta_force(
191 this->flow_control.get_beta());
192 }
193
194 // Calculate the void fraction and evaluate it within the matrix-free
195 // operator
196 {
197 TimerOutput::Scope t(this->computing_timer, "Calculate void fraction");
198 void_fraction_manager.calculate_void_fraction(
199 this->simulation_control->get_current_time());
200
201 // The base matrix-free operator is not aware of the void fraction. We
202 // must do a cast here to ensure that the operator is of the right type
203 if (auto mf_operator = dynamic_cast<VANSOperator<dim, double> *>(
204 this->system_operator.get()))
205 mf_operator->evaluate_void_fraction(void_fraction_manager);
206 }
207
208 this->iterate();
209 this->postprocess(false);
210 this->finish_time_step();
211
212 if (this->simulation_parameters.timer.type ==
213 Parameters::Timer::Type::iteration)
214 this->print_mg_setup_times();
215 }
216
217 if (this->simulation_parameters.timer.type == Parameters::Timer::Type::end)
218 this->print_mg_setup_times();
219
220 this->finish_simulation();
221}
