29 /**
30  * @brief Class that stores the information required by the assembly procedure
31  * for a Tracer advection-diffusion equation. Consequently, this class
32  *calculates the tracer (values, gradients, laplacians) and the shape function
33  * (values, gradients, laplacians) at all the gauss points for all degrees
34  * of freedom and stores it into arrays. Additionnaly, the use can request
35  * that this class gathers additional fields for physics which are coupled
36  * to the Tracer equation, such as the velocity which is required. This class
37  * serves as a seperation between the evaluation at the gauss point of the
38  * variables of interest and their use in the assembly, which is carried out
39  * by the assembler functions. For more information on this design, the reader
40  * can consult deal.II step-9
41  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
42  * example, the scratch is a struct instead of a templated class because of the
43  * simplicity of step-9.
44  *
45  * @tparam dim An integer that denotes the dimension of the space in which
46  * the flow is solved
47  * @ingroup solvers
48  **/

138  /** @brief Reinitializes the content of the scratch.
139  *
140  * Using the FeValues and the content of the solutions and previous solutions,
141  * fills all of the class member of the scratch.
142  *
143  * @param[in] cell The cell over which the assembly is being carried.
144  * This cell must be compatible with the fe which is used to fill the
145  * FeValues.
146  *
147  * @param[in] current_solution The present value of the solution.
148  *
149  * @param[in] previous_solutions The solutions at the previous time steps.
150  *
151  * @param[in] source_function The function describing the tracer source term.
152  *
153  * @param[in] levelset_function The function describing the particles (if
154  * there are any).
155  */
156  template <typename VectorType>
157  void
158  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
159  const VectorType &current_solution,
160  const std::vector<VectorType> &previous_solutions,
161  const Function<dim> *source_function,
162  const Function<dim> *levelset_function)
163  {
164  this->fe_values_tracer.reinit(cell);
165 
166  quadrature_points = this->fe_values_tracer.get_quadrature_points();
167  auto &fe_tracer = this->fe_values_tracer.get_fe();
168 
169  source_function->value_list(quadrature_points, source);
170 
171  if (properties_manager.field_is_required(field::levelset))
172  {
173  Assert(
174  levelset_function != nullptr,
175  ExcMessage(
176  "Levelset function is required for tracer assembly, but the level set function is a nullptr"));
177 
178  levelset_function->value_list(quadrature_points, sdf_values);
179  }
180 
181  // Compute cell diameter
182  double cell_measure =
183  compute_cell_measure_with_JxW(this->fe_values_tracer.get_JxW_values());
184  this->cell_size =
185  compute_cell_diameter<dim>(cell_measure, fe_tracer.degree);
186 
187  // Gather tracer (values, gradient and laplacian)
188  this->fe_values_tracer.get_function_values(current_solution,
189  this->tracer_values);
190  this->fe_values_tracer.get_function_gradients(current_solution,
191  this->tracer_gradients);
192  this->fe_values_tracer.get_function_laplacians(current_solution,
193  this->tracer_laplacians);
194  this->fe_values_tracer.get_function_gradients(
195  previous_solutions[0], this->previous_tracer_gradients);
196 
197  // Gather previous tracer values
198  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
199  {
200  this->fe_values_tracer.get_function_values(previous_solutions[p],
201  previous_tracer_values[p]);
202  }
203 
204  for (unsigned int q = 0; q < n_q_points; ++q)
205  {
206  this->JxW[q] = this->fe_values_tracer.JxW(q);
207 
208  for (unsigned int k = 0; k < n_dofs; ++k)
209  {
210  // Shape function
211  this->phi[q][k] = this->fe_values_tracer.shape_value(k, q);
212  this->grad_phi[q][k] = this->fe_values_tracer.shape_grad(k, q);
213  this->hess_phi[q][k] = this->fe_values_tracer.shape_hessian(k, q);
214  this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
215  }
216  }
217 
218  boundary_index = 0;
219  }

223  /** @brief Reinitialize the velocity, calculated by the fluid dynamics while also taking into account ALE
224  *
225  * @tparam VectorType The Vector type used for the solvers
226  *
227  * @param cell The cell for which the velocity is reinitialized
228  * This cell must be compatible with the Fluid Dynamics FE
229  *
230  * @param current_solution The present value of the solution for [u,p]
231  *
232  * @param ale The ALE parameters which include the ALE function
233  *
234  * @param drift_velocity Function to calculate the drift velocity
235  *
236  */
237 
238  template <typename VectorType>
239  void
240  reinit_velocity(
241  const typename DoFHandler<dim>::active_cell_iterator &cell,
242  const VectorType &velocity_solution,
243  const Parameters::ALE<dim> &ale,
244  std::shared_ptr<Functions::ParsedFunction<dim>> drift_velocity)
245  {
246  this->fe_values_fd.reinit(cell);
247 
248  this->fe_values_fd[velocities].get_function_values(velocity_solution,
249  velocity_values);
250 
251  // Add the drift velocity to the velocity to account for tracer drift flux
252  // modeling
253  Tensor<1, dim> drift_velocity_tensor;
254  Vector<double> drift_velocity_vector(dim);
255 
256  for (unsigned int q = 0; q < n_q_points; ++q)
257  {
258  drift_velocity->vector_value(quadrature_points[q],
259  drift_velocity_vector);
260  for (unsigned int d = 0; d < dim; ++d)
261  drift_velocity_tensor[d] = drift_velocity_vector[d];
262 
263  velocity_values[q] += drift_velocity_tensor;
264  }
265 
266  if (!ale.enabled())
267  return;
268 
269  // ALE enabled, so extract the ALE velocity and subtract it from the
270  // velocity obtained from the fluid dynamics
271  Tensor<1, dim> velocity_ale;
272  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
273  ale.velocity;
274  Vector<double> velocity_ale_vector(dim);
275 
276  for (unsigned int q = 0; q < n_q_points; ++q)
277  {
278  velocity_ale_function->vector_value(quadrature_points[q],
279  velocity_ale_vector);
280  for (unsigned int d = 0; d < dim; ++d)
281  velocity_ale[d] = velocity_ale_vector[d];
282 
283  velocity_values[q] -= velocity_ale;
284  }
285  }
