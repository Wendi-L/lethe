278 /**
279  * @brief Class that assembles the drag force using the Koch and Hill drag model for the
280  * VANS equations where the momentum exchange coefficient
281  * beta = ((18 * mu * cell_void_fraction^2 *
282  (1 - cell_void_fraction)) / pow(dp, 2)) *
283  (f0 + 0.5 * f3 * cell_void_fraction * re) *
284  Vp /(1 - cell_void_fraction) where f0 and f3 are functions given by:
285  if ((1 - cell_void_fraction) < 0.4)
286  {
287  f0 = (1 + 3 * sqrt((1 - cell_void_fraction) / 2) +
288  (135.0 / 64) * (1 - cell_void_fraction) *
289  log(1 - cell_void_fraction) +
290  16.14 * (1 - cell_void_fraction)) /
291  (1 + 0.681 * (1 - cell_void_fraction) -
292  8.48 * (1 - cell_void_fraction)^2 +
293  8.14 * (1 - cell_void_fraction)^3);
294  }
295  else if ((1 - cell_void_fraction) >= 0.4)
296  {
297  f0 = 10 * (1 - cell_void_fraction) / cell_void_fraction^3;
298  }
299 
300  f3 = 0.0673 + 0.212 * (1 - cell_void_fraction) +
301  0.0232 / pow(cell_void_fraction, 5);
302  * @tparam dim An integer that denotes the number of spatial dimensions
303  *
304  * @ingroup assemblers
305  */

368 /**
369  * @brief Class that assembles the drag force using Gidaspow model for the
370  * VANS equations where the momentum_transfer_coefficient is calculated
371  according to the following Marchelli et al. (2020):
372  if (cell_void_fraction > 0.8)
373  {
374  momentum_transfer_coefficient =
375  (18 * pow(cell_void_fraction, -3.65) *
376  (1 + 0.15 * pow(Re_p[particle_number], 0.687))) *
377  (particle_properties[DEM::PropertiesIndex::mass] * viscosity /
378  (pow(particle_properties[DEM::PropertiesIndex::dp], 2) *
379  particle_density));
380  }
381  else
382  {
383  // Assuming the sphericity of particles = 1
384  momentum_transfer_coefficient =
385  (150 * (1 - cell_void_fraction) / pow(cell_void_fraction, 2) +
386  1.75 * Re_p[particle_number] / pow(cell_void_fraction, 2)) *
387  (particle_properties[DEM::PropertiesIndex::mass] * viscosity /
388  (pow(particle_properties[DEM::PropertiesIndex::dp], 2) *
389  particle_density));
390  }
