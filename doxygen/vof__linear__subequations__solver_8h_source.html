40  /**
41  * @brief Constructor for linear subequations solvers within the VOF
42  * auxiliary physics
43  *
44  * @param[in] p_subequation_id Identifier corresponding to the subequation.
45  *
46  * @param[in,out] p_subequations Subequations interface object used to get
47  * information from other subequations and store information from the current
48  * one.
49  *
50  * @param[in] p_multiphysics Multiphysics interface object used to get
51  * information from physics.
52  *
53  * @param[in] p_simulation_parameters Simulation parameters.
54  *
55  * @param[in] p_triangulation Distributed mesh information.
56  *
57  * @param[in] p_simulation_control Object responsible for the control of
58  * steady-state and transient simulations. Contains all the information
59  * related to time stepping and the stopping criteria.
60  *
61  * @param[in] p_subequation_verbosity Parameter indicating the verbosity level
62  * of the solver.
63  *
64  * @param[in] p_pcout Parallel cout used to print the information.
65  */
66  VOFLinearSubequationsSolver(
67  const VOFSubequationsID &p_subequation_id,
68  VOFSubequationsInterface<dim> *p_subequations,
69  MultiphysicsInterface<dim> *p_multiphysics,
70  const SimulationParameters<dim> &p_simulation_parameters,
71  std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
72  &p_triangulation,
73  std::shared_ptr<SimulationControl> &p_simulation_control,
74  const Parameters::Verbosity &p_subequation_verbosity,
75  const ConditionalOStream &p_pcout)
76  : PhysicsLinearSubequationsSolver(p_pcout)
77  , subequation_id(p_subequation_id)
78  , subequations(p_subequations)
79  , multiphysics(p_multiphysics)
80  , simulation_parameters(p_simulation_parameters)
81  , triangulation(p_triangulation)
82  , simulation_control(p_simulation_control)
83  , dof_handler(*triangulation)
84  , linear_solver_verbosity(
85  p_simulation_parameters.linear_solver.at(PhysicsID::VOF).verbosity)
86  , subequation_verbosity(p_subequation_verbosity)
87  {
88  if (this->simulation_parameters.mesh.simplex)
89  {
90  // for simplex meshes
91  const FE_SimplexP<dim> subequation_fe(
92  this->simulation_parameters.fem_parameters.VOF_order);
93  this->fe = std::make_shared<FESystem<dim>>(subequation_fe, dim);
94  this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
95 
96  this->cell_quadrature =
97  std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
98  }
99  else
100  {
101  // Usual case, for quad/hex meshes
102  const FE_Q<dim> subequation_fe(
103  this->simulation_parameters.fem_parameters.VOF_order);
104  this->fe = std::make_shared<FESystem<dim>>(subequation_fe, dim);
105  this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
106 
107  this->cell_quadrature =
108  std::make_shared<QGauss<dim>>(this->fe->degree + 1);
109  }
110  }
